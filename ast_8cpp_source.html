86/// Ignore any `Separator` tokens if any are present.
87/// \returns true if a separator was encountered (and consumed)
88auto ignore_separator(TokenIterator& tokens,
89 [[maybe_unused]] const source_location location = source_location::current()) -> bool {
90 bool found_separator = false;
91 while (!tokens.at_end() && tokens->m_type == Separator) {
92#ifdef YUME_SPEW_CONSUMED_TOKENS
93 std::cerr << "consumed " << *tokens << " at " << at(location) << "\n";
94#endif
95 ++tokens;
96 found_separator = true;
97 }
98 return found_separator;
99}

101/// If the next token doesn't have the type, `token_type`, throw a runtime exception.
102void expect(TokenIterator& tokens, Token::Type token_type,
103 const source_location location = source_location::current()) {
104 if (tokens->m_type != token_type) {
105 throw std::runtime_error("Expected token type "s + Token::type_name(token_type) + ", got " + to_string(*tokens) +
106 " at " + at(location));
107 }
108}

110/// Consume all subsequent `Separator` tokens. Throws if none were found.
111void require_separator(TokenIterator& tokens, const source_location location = source_location::current()) {
112 expect(tokens, Separator, location);
113 ignore_separator(tokens, location);
114}

116/// Consume a token of type `token_type` with the given `payload`. Throws if it wasn't encountered.
117void consume(TokenIterator& tokens, Token::Type token_type, Atom payload,
118 const source_location location = source_location::current()) {
119 ignore_separator(tokens);
120 expect(tokens, token_type, location);
121 if (tokens->m_payload != payload) {
122 throw std::runtime_error("Expected payload atom "s + string(payload) + ", got " + to_string(*tokens) + " at " +
123 at(location));
124 }
125
126#ifdef YUME_SPEW_CONSUMED_TOKENS
127 std::cerr << "consume: " << *tokens << " at " << at(location) << "\n";
128#endif
129
130 tokens++;
131}

133/// Attempt to consume a token of type `token_type` with the given `payload`. Does nothing if it wasn't encountered.
134auto try_consume(TokenIterator& tokens, Token::Type tokenType, Atom payload,
135 [[maybe_unused]] const source_location location = source_location::current()) -> bool {
136 if (tokens->m_type != tokenType || tokens->m_payload != payload) {
137 return false;
138 }
139
140#ifdef YUME_SPEW_CONSUMED_TOKENS
141 std::cerr << "try_consume: " << *tokens << " at " << at(location) << "\n";
142#endif
143
144 tokens++;
145 return true;
146}

148/// Check if the token ahead by `ahead` is of type `token_type` with the given `payload`.
149auto try_peek(TokenIterator& tokens, int ahead, Token::Type token_type, Atom payload,
150 [[maybe_unused]] const source_location location = source_location::current()) -> bool {
151 auto token = tokens + ahead;
152
153#ifdef YUME_SPEW_CONSUMED_TOKENS
154 std::cerr << "try_peek ahead by " << ahead << ": expected " << Token::type_name(token_type) << " " << string(payload)
155 << ", got " << *token << " at " << at(location) << "\n";
156#endif
157
158 return !(token->m_type != token_type || token->m_payload != payload);
159}

161/// Check if the token ahead by `ahead` is of type `token_type`.
162auto try_peek(TokenIterator& tokens, int ahead, Token::Type token_type,
163 [[maybe_unused]] const source_location location = source_location::current()) -> bool {
164 auto token = tokens + ahead;
165
166#ifdef YUME_SPEW_CONSUMED_TOKENS
167 std::cerr << "try_peek ahead by " << ahead << ": expected " << Token::type_name(token_type) << ", got " << *token
168 << " at " << at(location) << "\n";
169#endif
170
171 return token->m_type == token_type;

174/// Consume tokens until a token of type `token_type` with the given `payload` is encountered.
175/// `action` (a no-arg function) is called every time. Between each call, a comma is expected.
176void consume_with_commas_until(TokenIterator& tokens, Token::Type token_type, Atom payload, auto action,
177 const source_location location = source_location::current()) {
178 int i = 0;
179 while (!try_consume(tokens, token_type, payload, location)) {
180 if (i++ > 0) {
181 consume(tokens, Symbol, SYM_COMMA, location);
182 }
183 action();
184 }
185}

197/// Return the payload of the next token. Throws if the next token isn't a `Word`.
198auto consume_word(TokenIterator& tokens, const source_location location = source_location::current()) -> string {
199 ignore_separator(tokens);
200 if (tokens->m_type != Word) {
201 throw std::runtime_error("Expected word, got "s + to_string(*tokens) + " at " + at(location));
202 }
203 return *next(tokens, location).m_payload;
204}

206/// Check if the string begins with a capital letter. Used for types, as all types must be capitalized.
207auto is_uword(const string& word) -> bool { return isupper(word.front()) != 0; }

209/// Check if the ahead by `ahead` is a capitalized word.
210auto try_peek_uword(TokenIterator& tokens, int ahead,
211 [[maybe_unused]] const source_location location = source_location::current()) -> bool {
212 auto token = tokens + ahead;
213
214#ifdef YUME_SPEW_CONSUMED_TOKENS
215 std::cerr << "try_peek ahead by " << ahead << ": expected uword, got " << *token << " at " << at(location) << "\n";
216#endif
217
218 return token->m_type == Word && is_uword(token->m_payload.value());
219}

227auto Program::parse(TokenIterator& tokens) -> unique_ptr<Program> {
228 ignore_separator(tokens);
229 auto entry = tokens.begin();
230
231 auto statements = vector<unique_ptr<Stmt>>{};
232 while (!tokens.at_end()) {
233 statements.push_back(parse_stmt(tokens));
234 }
235
236 return std::make_unique<Program>(ts(entry, tokens.begin()), move(statements));
237}

239static auto parse_fn_name(TokenIterator& tokens) -> string {
240 string name{};
241 if (tokens->m_type == Word) {
242 name = consume_word(tokens);
243 } else if (tokens->m_type == Symbol) {
244 // Try to parse an operator name, as in `def +()`
245 bool found_op = false;
246 for (const auto& op_row : operators()) {
247 for (const auto& op : op_row) {
248 if (try_consume(tokens, Symbol, op)) {
249 found_op = true;
250 name = op;
251 break;
252 }
253 }
254 if (found_op) {
255 break;
256 }
257 }
258
259 // If an operator wasn't found, try parse the operator []
260 if (try_consume(tokens, Symbol, SYM_LBRACKET)) {
261 consume(tokens, Symbol, SYM_RBRACKET);
262 name = "[]";
263 } else if (try_consume(tokens, Symbol, SYM_BANG)) {
264 name = "!"; // ! is unary, but the above operator check only checked binary ones
265 }
266 }
267
268 // Check if an equal sign follows, for fused assignement operators such as `+=` or `[]=`
269 if (try_consume(tokens, Symbol, SYM_EQ)) {
270 name += "=";
271 }
272
273 return name;
274}

276static auto parse_struct_decl(TokenIterator& tokens) -> unique_ptr<StructDecl> {
277 auto entry = tokens.begin();
278
279 consume(tokens, Word, KWD_STRUCT);
280 const string name = consume_word(tokens);
281 if (!(is_uword(name))) {
282 throw std::runtime_error("Expected capitalized name for struct decl");
283 }
284 auto type_args = vector<string>{};
285 if (try_consume(tokens, Symbol, SYM_LT)) {
286 consume_with_commas_until(tokens, Symbol, SYM_GT, [&] { type_args.push_back(consume_word(tokens)); });
287 }
288
289 consume(tokens, Symbol, SYM_LPAREN);
290 auto fields = vector<TypeName>{};
291 consume_with_commas_until(tokens, Symbol, SYM_RPAREN, [&] { fields.push_back(std::move(*parse_type_name(tokens))); });
292
293 auto body = vector<unique_ptr<Stmt>>{};
294 auto body_begin = entry;
295
296 require_separator(tokens);
297
298 body_begin = tokens.begin();
299 while (!try_consume(tokens, Word, KWD_END)) {
300 body.push_back(parse_stmt(tokens));
301 ignore_separator(tokens);
302 }
303
304 return std::make_unique<StructDecl>(ts(entry, tokens.begin()), name, move(fields), type_args,
305 Compound(ts(body_begin, tokens.begin()), move(body)));
306}

308static auto parse_fn_decl(TokenIterator& tokens) -> unique_ptr<FnDecl> {
309 auto entry = tokens.begin();
310
311 consume(tokens, Word, KWD_DEF);
312 const string name = parse_fn_name(tokens);
313 auto type_args = vector<string>{};
314 if (try_consume(tokens, Symbol, SYM_LT)) {
315 consume_with_commas_until(tokens, Symbol, SYM_GT, [&] { type_args.push_back(consume_word(tokens)); });
316 }
317 consume(tokens, Symbol, SYM_LPAREN);
318
319 auto args = vector<unique_ptr<TypeName>>{};
320 consume_with_commas_until(tokens, Symbol, SYM_RPAREN, [&] { args.push_back(parse_type_name(tokens)); });
321
322 auto return_type = try_parse_type(tokens);
323 auto body = vector<unique_ptr<Stmt>>{};
324 auto body_begin = entry;
325
326 if (try_consume(tokens, Symbol, SYM_EQ)) { // A "short" function definition, consists of a single expression
327 if (try_consume(tokens, Word, KWD_PRIMITIVE)) {
328 consume(tokens, Symbol, SYM_LPAREN);
329 auto primitive = consume_word(tokens);
330 consume(tokens, Symbol, SYM_RPAREN);
331 auto varargs = try_consume(tokens, Word, KWD_VARARGS);
332 return std::make_unique<FnDecl>(ts(entry, tokens.begin()), name, move(args), type_args, move(return_type),
333 varargs, primitive);
334 }
335 body_begin = tokens.begin();
336 auto expr = parse_expr(tokens);
337 body.push_back(std::make_unique<ReturnStmt>(ts(entry, tokens.begin()), move(expr)));
338 } else {
339 require_separator(tokens);
340
341 body_begin = tokens.begin();
342 while (!try_consume(tokens, Word, KWD_END)) {
343 body.push_back(parse_stmt(tokens));
344 ignore_separator(tokens);
345 }
346 }
347
348 return std::make_unique<FnDecl>(ts(entry, tokens.begin()), name, move(args), type_args, move(return_type),
349 std::make_unique<Compound>(ts(body_begin, tokens.begin()), move(body)));
350}

366static auto parse_while_stmt(TokenIterator& tokens) -> unique_ptr<WhileStmt> {
367 auto entry = tokens.begin();
368
369 consume(tokens, Word, KWD_WHILE);
370 auto cond = parse_expr(tokens);
371
372 ignore_separator(tokens);
373
374 auto body_begin = tokens.begin();
375 auto body = vector<unique_ptr<Stmt>>{};
376 while (!try_consume(tokens, Word, KWD_END)) {
377 body.push_back(parse_stmt(tokens));
378 ignore_separator(tokens);
379 }
380
381 auto compound = std::make_unique<Compound>(ts(body_begin, tokens.begin()), move(body));
382
383 return std::make_unique<WhileStmt>(ts(entry, tokens.begin()), move(cond), move(compound));
384}

386static auto parse_return_stmt(TokenIterator& tokens) -> unique_ptr<ReturnStmt> {
387 auto entry = tokens.begin();
388
389 consume(tokens, Word, KWD_RETURN);
390 if (!tokens.at_end() && try_peek(tokens, 0, Separator)) {
391 return std::make_unique<ReturnStmt>(ts(entry, tokens.begin()), optional<unique_ptr<Expr>>{});
392 }
393 auto expr = parse_expr(tokens);
394
395 return std::make_unique<ReturnStmt>(ts(entry, tokens.begin()), optional<unique_ptr<Expr>>{move(expr)});
396}

398static auto parse_if_stmt(TokenIterator& tokens) -> unique_ptr<IfStmt> {
399 auto entry = tokens.begin();
400 auto clause_begin = entry;
401 consume(tokens, Word, KWD_IF);
402 auto cond = parse_expr(tokens);
403 if (!try_consume(tokens, Word, KWD_THEN)) {
404 require_separator(tokens);
405 }
406
407 auto current_entry = tokens.begin();
408 auto else_entry = tokens.begin();
409 auto clauses = vector<unique_ptr<IfClause>>{};
410 auto current_body = vector<unique_ptr<Stmt>>{};
411 auto else_body = vector<unique_ptr<Stmt>>{};
412 bool in_else = false;
413
414 while (true) {
415 auto current_clause_begin = tokens.begin();
416 if (try_consume(tokens, Word, KWD_END)) {
417 break;
418 }
419 if (try_consume(tokens, Word, KWD_ELSE)) {
420 // An `else` followed by an `if` begins a new clause of the same if statement.
421 if (!in_else && try_consume(tokens, Word, KWD_IF)) {
422 clauses.push_back(std::make_unique<IfClause>(
423 ts(clause_begin, tokens.begin()), move(cond),
424 std::make_unique<Compound>(ts(current_entry, tokens.begin()), move(current_body))));
425 current_body = vector<unique_ptr<Stmt>>{};
426 cond = parse_expr(tokens);
427 current_entry = tokens.begin();
428 clause_begin = current_clause_begin;
429 } else {
430 in_else = true;
431 else_entry = tokens.begin();
432 }
433 if (!try_consume(tokens, Word, KWD_THEN)) {
434 require_separator(tokens);
435 }
436 }
437 auto st = parse_stmt(tokens);
438 if (in_else) {
439 else_body.push_back(move(st));
440 } else {
441 current_body.push_back(move(st));
442 }
443 }
444
445 if (else_body.empty()) {
446 else_entry = tokens.begin();
447 }
448
449 clauses.push_back(
450 std::make_unique<IfClause>(ts(clause_begin, else_entry - 1), move(cond),
451 std::make_unique<Compound>(ts(current_entry, else_entry - 1), move(current_body))));
452
453 auto else_clause = optional<unique_ptr<Compound>>{};
454 if (!else_body.empty()) {
455 else_clause = std::make_unique<Compound>(ts(else_entry, tokens.begin()), move(else_body));
456 }
457
458 return std::make_unique<IfStmt>(ts(entry, tokens.begin()), move(clauses), move(else_clause));
459}

485static auto parse_primary(TokenIterator& tokens) -> unique_ptr<Expr> {
486 auto entry = tokens.begin();
487 if (try_consume(tokens, Symbol, SYM_LPAREN)) {
488 auto val = parse_expr(tokens);
489 consume(tokens, Symbol, SYM_RPAREN);
490 return val;
491 }
492 if (tokens->m_type == Number) {
493 return parse_number_expr(tokens);
494 }
495 if (tokens->m_type == Token::Type::Literal) {
496 return parse_string_expr(tokens);
497 }
498 if (tokens->m_type == Token::Type::Char) {
499 return parse_char_expr(tokens);
500 }
501 if (try_consume(tokens, Word, KWD_TRUE)) {
502 return std::make_unique<BoolExpr>(ts(entry, tokens.begin()), true);
503 }
504 if (try_consume(tokens, Word, KWD_FALSE)) {
505 return std::make_unique<BoolExpr>(ts(entry, tokens.begin()), false);
506 }
507 if (tokens->m_type == Word) {
508 bool use_self = false;
509 if (try_peek_uword(tokens, 0) || try_peek(tokens, 0, Word, KWD_SELF)) {
510 auto type = parse_type(tokens);
511 if (try_consume(tokens, Symbol, SYM_LPAREN)) {
512 auto call_args = vector<unique_ptr<Expr>>{};
513 consume_with_commas_until(tokens, Symbol, SYM_RPAREN, [&] { call_args.push_back(parse_expr(tokens)); });
514 return std::make_unique<CtorExpr>(ts(entry, tokens.begin()), move(type), move(call_args));
515 }
516 if (try_consume(tokens, Symbol, SYM_LBRACKET)) {
517 auto slice_members = vector<unique_ptr<Expr>>{};
518 consume_with_commas_until(tokens, Symbol, SYM_RBRACKET, [&] { slice_members.push_back(parse_expr(tokens)); });
519 return std::make_unique<SliceExpr>(ts(entry, tokens.begin()), move(type), move(slice_members));
520 }
521 // `self` may be used as a type in a constructor `self()`, but also by itself as a variable.
522 if (llvm::isa<SelfType>(*type)) {
523 use_self = true;
524 } else {
525 throw std::runtime_error("Couldn't make an expression from here with an uword");
526 }
527 }
528 auto name = use_self ? "self" : consume_word(tokens);
529 if (try_consume(tokens, Symbol, SYM_LPAREN)) {
530 auto call_args = vector<unique_ptr<Expr>>{};
531 consume_with_commas_until(tokens, Symbol, SYM_RPAREN, [&] { call_args.push_back(parse_expr(tokens)); });
532 return std::make_unique<CallExpr>(ts(entry, tokens.begin()), name, move(call_args));
533 }
534 return std::make_unique<VarExpr>(ts(entry, 1), name);
535 }
536 throw std::runtime_error("Couldn't make an expression from here");
537}

539static auto parse_receiver(TokenIterator& tokens, unique_ptr<Expr> receiver, auto receiver_entry) -> unique_ptr<Expr> {
540 auto entry = tokens.begin();
541 if (try_consume(tokens, Symbol, SYM_DOT)) {
542 auto name = consume_word(tokens);
543 auto call_args = vector<unique_ptr<Expr>>{};
544 call_args.push_back(move(receiver));
545 if (try_consume(tokens, Symbol, SYM_LPAREN)) { // A call with a dot `a.b(...)`
546 consume_with_commas_until(tokens, Symbol, SYM_RPAREN, [&] { call_args.push_back(parse_expr(tokens)); });
547 auto call = std::make_unique<CallExpr>(ts(entry + 1, tokens.begin()), name, move(call_args));
548 return parse_receiver(tokens, move(call), receiver_entry);
549 }
550 if (try_consume(tokens, Symbol, SYM_EQ)) { // A setter `a.b = ...`
551 auto value = parse_expr(tokens);
552 call_args.push_back(move(value));
553 auto call = std::make_unique<CallExpr>(ts(entry + 1, tokens.begin()), name + '=', move(call_args));
554 return parse_receiver(tokens, move(call), receiver_entry);
555 }
556 auto noarg_call = std::make_unique<CallExpr>(ts(receiver_entry, tokens.begin()), name, move(call_args));
557 return parse_receiver(tokens, move(noarg_call), receiver_entry);
558 }
559 if (try_consume(tokens, Symbol, SYM_EQ)) {
560 auto value = parse_expr(tokens);
561 auto assign = std::make_unique<AssignExpr>(ts(receiver_entry, tokens.begin()), move(receiver), move(value));
562 return parse_receiver(tokens, move(assign), receiver_entry);
563 }
564 if (try_consume(tokens, Symbol, SYM_LBRACKET)) {
565 auto args = vector<unique_ptr<Expr>>{};
566 args.push_back(move(receiver));
567 args.push_back(parse_expr(tokens));
568 consume(tokens, Symbol, SYM_RBRACKET);
569 if (try_consume(tokens, Symbol, SYM_EQ)) {
570 auto value = parse_expr(tokens);
571 args.push_back(move(value));
572 auto call = std::make_unique<CallExpr>(ts(entry, tokens.begin()), "[]=", move(args));
573 return parse_receiver(tokens, move(call), receiver_entry);
574 }
575 auto call = std::make_unique<CallExpr>(ts(entry, tokens.begin()), "[]", move(args));
576 return parse_receiver(tokens, move(call), receiver_entry);
577 }
578 if (try_consume(tokens, Symbol, SYM_COLON)) {
579 consume(tokens, Symbol, SYM_COLON);
580 auto field = consume_word(tokens);
581 auto access = std::make_unique<FieldAccessExpr>(ts(receiver_entry, tokens.begin()), move(receiver), field);
582 return parse_receiver(tokens, move(access), receiver_entry);
583 }
584 return receiver;
585}

589 return parse_receiver(tokens, parse_primary(tokens), entry);

592static auto parse_unary(TokenIterator& tokens) -> unique_ptr<Expr> {
593 auto entry = tokens.begin();
594 for (const auto& un_op : unary_operators()) {
595 if (try_consume(tokens, Symbol, un_op)) {
596 auto value = parse_receiver(tokens);
597 auto args = vector<unique_ptr<Expr>>{};
598 args.push_back(move(value));
599 return std::make_unique<CallExpr>(ts(entry, tokens.begin()), un_op, move(args));
600 }
601 }
602 return parse_receiver(tokens);
603}

605static auto parse_operator(TokenIterator& tokens, size_t n = 0) -> unique_ptr<Expr> {
606 auto entry = tokens.begin();
607 const auto ops = operators();
608 if (n == ops.size()) {
609 return parse_unary(tokens);
610 }
611 auto left = parse_operator(tokens, n + 1);
612 while (true) {
613 auto found_operator = false;
614 for (const auto& op : ops[n]) {
615 if (try_consume(tokens, Symbol, op)) {
616 auto right = parse_operator(tokens, n + 1);
617 auto args = vector<unique_ptr<Expr>>{};
618 args.push_back(move(left));
619 args.push_back(move(right));
620 left = std::make_unique<CallExpr>(ts(entry, tokens.begin()), op, move(args));
621 found_operator = true;
622 break;
623 }
624 }
625 if (!found_operator) {
626 break;
627 }
628 }
629 return left;
630}

632static auto parse_stmt(TokenIterator& tokens) -> unique_ptr<Stmt> {
633 auto stat = unique_ptr<Stmt>();
634
635 if (tokens->is_keyword(KWD_DEF)) {
636 stat = parse_fn_decl(tokens);
637 } else if (tokens->is_keyword(KWD_STRUCT)) {
638 stat = parse_struct_decl(tokens);
639 } else if (tokens->is_keyword(KWD_LET)) {
640 stat = parse_var_decl(tokens);
641 } else if (tokens->is_keyword(KWD_WHILE)) {
642 stat = parse_while_stmt(tokens);
643 } else if (tokens->is_keyword(KWD_IF)) {
644 stat = parse_if_stmt(tokens);
645 } else if (tokens->is_keyword(KWD_RETURN)) {
646 stat = parse_return_stmt(tokens);
647 } else {
648 stat = parse_expr(tokens);
649 }
650
651 require_separator(tokens);
652 return stat;
653}

655static auto parse_type(TokenIterator& tokens) -> unique_ptr<Type> {
656 auto entry = tokens.begin();
657 auto base = [&]() -> unique_ptr<Type> {
658 if (try_consume(tokens, Word, KWD_SELF)) {
659 return std::make_unique<SelfType>(ts(entry, tokens.begin()));
660 }
661 const string name = consume_word(tokens);
662 if (!(is_uword(name))) {
663 throw std::runtime_error("Expected capitalized payload for simple type");
664 }
665
666 return std::make_unique<SimpleType>(ts(entry, tokens.begin()), name);
667 }();
668 while (true) {
669 if (try_consume(tokens, Word, KWD_PTR)) {
670 base = std::make_unique<QualType>(ts(entry, tokens.begin()), move(base), Qualifier::Ptr);
671 } else if (try_consume(tokens, Word, KWD_MUT)) {
672 base = std::make_unique<QualType>(ts(entry, tokens.begin()), move(base), Qualifier::Mut);
673 } else if (try_peek(tokens, 0, Symbol, SYM_LBRACKET) && try_peek(tokens, 1, Symbol, SYM_RBRACKET)) {
674 // Don't consume the `[` unless the `]` is directly after; it might be a slice literal.
675 consume(tokens, Symbol, SYM_LBRACKET);
676 consume(tokens, Symbol, SYM_RBRACKET);
677 base = std::make_unique<QualType>(ts(entry, tokens.begin()), move(base), Qualifier::Slice);
678 } else {
679 break;
680 }
681 }
682
683 return base;
684}

686static auto try_parse_type(TokenIterator& tokens) -> optional<unique_ptr<Type>> {
687 auto entry = tokens.begin();
688 if (tokens->m_type != Word || !tokens->m_payload.has_value()) {
689 return {};
690 }
691
692 const string name = consume_word(tokens);
693 if (make_atom(name) != KWD_SELF && !is_uword(name)) {
694 return {};
695 }
696 auto base = [&]() -> unique_ptr<Type> {
697 if (make_atom(name) == KWD_SELF) {
698 return std::make_unique<SelfType>(ts(entry, tokens.begin()));
699 }
700
701 return std::make_unique<SimpleType>(ts(entry, tokens.begin()), name);
702 }();
703
704 while (true) {
705 if (try_consume(tokens, Word, KWD_PTR)) {
706 base = std::make_unique<QualType>(ts(entry, tokens.begin()), move(base), Qualifier::Ptr);
707 } else if (try_consume(tokens, Word, KWD_MUT)) {
708 base = std::make_unique<QualType>(ts(entry, tokens.begin()), move(base), Qualifier::Mut);
709 } else if (try_peek(tokens, 0, Symbol, SYM_LBRACKET) && try_peek(tokens, 1, Symbol, SYM_RBRACKET)) {
710 // Don't consume the `[` unless the `]` is directly after; it might be a slice literal.
711 consume(tokens, Symbol, SYM_LBRACKET);
712 consume(tokens, Symbol, SYM_RBRACKET);
713 base = std::make_unique<QualType>(ts(entry, tokens.begin()), move(base), Qualifier::Slice);
714 } else {
715 break;
716 }
717 }
718
719 return base;
720}
