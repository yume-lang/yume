138/// Represents "any" kind of ast node of type `T`, or the absence of one. See `OptionalExpr` and `OptionalType`.
139/**
140 * This template exists to avoid passing around `unique_ptr<T>`s around in code that deals with AST nodes containing
141 * other generic nodes, such as `Compound`.
142 * This class has the exact same layout at `AnyBase`, but is nominally explicit in its semantics, in that it may be
143 * null. It replaces the earlier usages of `optional<unique_ptr<T>>` which in theory wastes memory for the "optional"
144 * part when a nullptr already expresses the desired semantics.
145 * Unlike `AnyBase`, this class does not perform null pointer checks, and may be default constructed, or constructed
146 * from a `nullptr`.
147 */
148template <typename T> class OptionalAnyBase {
149 unique_ptr<T> m_val;
150 friend AnyBase<T>;
151
152public:
153 OptionalAnyBase() : m_val() {}
154 explicit OptionalAnyBase(T* raw_ptr) : m_val{raw_ptr} {}
155 template <std::convertible_to<unique_ptr<T>> U> OptionalAnyBase(U uptr) : m_val{move(uptr)} {}
156 OptionalAnyBase(std::nullopt_t /* tag */) : m_val{} {}
157 explicit OptionalAnyBase(optional<unique_ptr<T>> opt_uptr)
158 : m_val{opt_uptr.has_value() ? move(*opt_uptr) : nullptr} {}
159
160 [[nodiscard]] auto operator->() const -> const T* { return m_val.operator->(); }
161 [[nodiscard]] auto operator*() const -> const T& { return *m_val; }
162 [[nodiscard]] auto operator->() -> T* { return m_val.operator->(); }
163 [[nodiscard]] auto operator*() -> T& { return *m_val; }
164
165 [[nodiscard]] operator bool() const { return static_cast<bool>(m_val); }
166 [[nodiscard]] auto has_value() const -> bool { return static_cast<bool>(m_val); }
167 [[nodiscard]] auto raw_ptr() const -> const T* { return m_val.get(); }
168 [[nodiscard]] auto raw_ptr() -> T* { return m_val.get(); }
169};

171/// Represents "any" kind of ast node of type `T`. See `AnyExpr`, `AnyStmt` and `AnyType`.
172/**
173 * This template exists to avoid passing around `unique_ptr<T>`s around in code that deals with AST nodes containing
174 * other generic nodes, such as `Compound`.
175 * This class also has strict nullptr checks when constructing, and cannot be default constructed. See

186 }
187 AnyBase(OptionalAnyBase<T>&& other) : Super(move(other.m_val)) {}
188
189 [[nodiscard]] auto operator->() const -> const T* { return Super::m_val.operator->(); }
190 [[nodiscard]] auto operator*() const -> const T& { return *Super::m_val; }

192 [[nodiscard]] auto operator*() -> T& { return *Super::m_val; }

210/// All nodes in the `AST` tree of the program inherit from this class.
211/**
212 * AST nodes cannot be copied or moved as all special member functions (aside from the destructor) are `delete`d. There
213 * will always be one instance for a specific node during the lifetype of the compiler.
214 * However note that AST nodes can be cloned with the `clone()` method, which will produce a deep copy. This is used
215 * when instantiating a template. The clone will have the exact same structure and locations but distinct types,
216 * depending on what the types the template is instantiated with.
217 */
218class AST {
219 /// Every subclass of `AST` has a distinct `Kind`.
220 const Kind m_kind;
221 /// The range of tokenizer `Token`s that this node was parsed from.
222 const span<Token> m_tok;
223 /// The value type of this node. Determined in the semantic phase; always empty after parsing.
224 optional<ty::Type> m_val_ty{};
225 /// \see Attachment
226 unique_ptr<Attachment> m_attach{std::make_unique<Attachment>()};
227
228protected:
229 /// Verify the type compatibility of the depends of this node, and merge the types if possible.
230 /// This is called every time the node's type is updated.
231 void unify_val_ty();
232
233 [[nodiscard]] auto tok() const noexcept -> span<Token> { return m_tok; }
234
235 AST(Kind kind, span<Token> tok) : m_kind(kind), m_tok(tok) {}
236
237public:
238 AST(const AST&) = delete;
239 AST(AST&&) = default;
240 auto operator=(const AST&) -> AST& = delete;
241 auto operator=(AST&&) -> AST& = delete;
242 virtual ~AST() = default;
243
244 /// Recursively visit this ast node and all its constituents. \see Visitor
245 virtual void visit(Visitor& visitor) const = 0;
246
247 [[nodiscard]] auto val_ty() const noexcept -> optional<ty::Type> { return m_val_ty; }
248 [[nodiscard]] auto ensure_ty() const -> ty::Type {
249 yume_assert(m_val_ty.has_value(), "Ensured that AST node has type, but one has not been assigned");
250 // NOLINTNEXTLINE(bugprone-unchecked-optional-access): clang-tidy doesn't accept yume_assert as an assertion
251 return *m_val_ty;
252 }
253 void val_ty(optional<ty::Type> type) {
254 m_val_ty = type;
255 for (auto* i : m_attach->observers) {
256 i->unify_val_ty();
257 }
258 }
259
260 /// Make the type of this node depend on the type of `other`.
261 /// \sa Attachment
262 void attach_to(nonnull<AST*> other) {
263 other->m_attach->observers.insert(this);
264 this->m_attach->depends.insert(other);
265 unify_val_ty();
266 }
267
268 [[nodiscard]] auto kind() const -> Kind { return m_kind; };
269 /// Human-readable string representation of the `Kind` of this node.
270 [[nodiscard]] auto kind_name() const -> string { return ast::kind_name(kind()); };
271 [[nodiscard]] auto token_range() const -> const span<Token>& { return m_tok; };
272
273 /// The union of the locations of the `Token`s making up this node.
274 [[nodiscard]] auto location() const -> Loc;
275
276 /// A short, string representation for debugging.
277 [[nodiscard]] virtual auto describe() const -> string;
278
279 /// Deep copy, except for the type and attachments.
280 [[nodiscard]] virtual auto clone() const -> AST* = 0;
281};
