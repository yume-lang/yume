98/// An iterator-like holding `Token`s, used when parsing.
99/**
100 * Every parse method usually takes this as the first parameter. This is its own struct as it actually holds two
101 * iterators, where one is the end. This is to provide safe indexing (avoiding going past the end) without having to
102 * pass the end iterator around as a separate parameter.
103 */
104struct TokenIterator {
105 VectorTokenIterator m_iterator;
106 VectorTokenIterator m_end;
107
108 constexpr TokenIterator(const VectorTokenIterator& iterator, const VectorTokenIterator& end)
109 : m_iterator{iterator}, m_end{end} {}
110
111 /// Check if the iterator is at the end and no more `Token`s could possibly be read.
112 [[nodiscard]] constexpr auto at_end() const noexcept -> bool { return m_iterator == m_end; }
113 [[nodiscard]] auto constexpr operator->() const noexcept -> Token* { return m_iterator.operator->(); }
114 [[nodiscard]] constexpr auto operator*() const noexcept -> Token { return m_iterator.operator*(); }
115 [[nodiscard]] constexpr auto operator+(int i) const noexcept -> TokenIterator {
116 return TokenIterator{m_iterator + i, m_end};
117 }
118 constexpr auto operator++() -> TokenIterator& {
119 if (at_end())
120 throw std::runtime_error("Can't increment past end");
121
122 ++m_iterator;
123 return *this;
124 }
125 auto operator++(int) -> TokenIterator {
126 if (at_end())
127 throw std::runtime_error("Can't increment past end");
128
129 return TokenIterator{m_iterator++, m_end};
130 }
131 /// Returns the underlying iterator. This shouldn't really be needed but I'm too lazy to properly model an iterator.
132 [[nodiscard]] auto begin() const -> VectorTokenIterator { return m_iterator; }
133};

137/// Represents the relationship between multiple `AST` nodes.
138/**
139 * `AST` nodes can be attached to propagate type information between them.
140 * TODO: describe type propagation more
141 */
142struct Attachment {
143 /// Observers are the "children" and should have their type updated when this node's type changes
144 mutable llvm::SmallPtrSet<const AST*, 2> observers{};
145 /// Depends are the "parents" and should be verified for type compatibility when this node's type changes
146 mutable llvm::SmallPtrSet<const AST*, 2> depends{};
147};
148
149/// All nodes in the `AST` tree of the program inherit from this class.
150/**
151 * AST nodes cannot be copied or moved as all special member functions (aside from the destructor) are `delete`d. There
152 * will always be one instance for a specific node during the lifetype of the compiler.
153 * However note that AST nodes can be cloned with the `clone()` method, which will produce a deep copy. This is used
154 * when instantiating a template. The clone will have the exact same structure and locations but distinct types,
155 * depending on what the types the template is instantiated with.
156 */
157class AST {
158 /// Every subclass of `AST` has a distinct `Kind`.
159 const Kind m_kind;
160 /// The range of tokenizer `Token`s that this node was parsed from.
161 const span<Token> m_tok;
162 /// The value type of this node. Determined in the semantic phase; always `nullptr` after parsing.
163 mutable const ty::Type* m_val_ty{};
164 /// \see Attachment
165 unique_ptr<Attachment> m_attach{std::make_unique<Attachment>()};
166
167protected:
168 /// Verify the type compatibility of the depends of this node, and merge the types if possible.
169 /// This is called every time the node's type is updated.
170 void unify_val_ty() const {
171 for (const auto* other : m_attach->depends) {
172 if (m_val_ty == other->m_val_ty || other->m_val_ty == nullptr)
173 return;
174
175 if (m_val_ty == nullptr) {
176 m_val_ty = other->m_val_ty;
177 } else {
178 const auto* merged = m_val_ty->coalesce(*other->m_val_ty);
179 if (merged == nullptr) {
180 throw std::logic_error("Conflicting types between AST nodes that are attached: `"s + m_val_ty->name() +
181 "` vs `" + other->m_val_ty->name() + "`!");
182 }
183 m_val_ty = merged;
184 }
185 }
186 }
187
188 [[nodiscard]] auto tok() const noexcept -> span<Token> { return m_tok; }
189
190 AST(Kind kind, span<Token> tok) : m_kind(kind), m_tok(tok) {}
191
192public:
193 AST(const AST&) = delete;
194 AST(AST&&) = default;
195 auto operator=(const AST&) -> AST& = delete;
196 auto operator=(AST&&) -> AST& = delete;
197 virtual ~AST() = default;
198
199 virtual void visit(Visitor& visitor) = 0;
200
201 [[nodiscard]] auto val_ty() const noexcept -> const ty::Type* { return m_val_ty; }
202 void val_ty(const ty::Type* type) const {
203 m_val_ty = type;
204 for (const auto* i : m_attach->observers) {
205 i->unify_val_ty();
206 }
207 }
208
209 /// Make the type of this node depend on the type of `other`.
210 /// \sa Attachment
211 void attach_to(const AST* other) const {
212 other->m_attach->observers.insert(this);
213 this->m_attach->depends.insert(other);
214 unify_val_ty();
215 }
216
217 [[nodiscard]] auto kind() const -> Kind { return m_kind; };
218 /// Human-readable string representation of the `Kind` of this node.
219 [[nodiscard]] auto kind_name() const -> string { return ast::kind_name(kind()); };
220 [[nodiscard]] auto token_range() const -> const span<Token>& { return m_tok; };
221
222 /// The union of the locations of the `Token`s making up this node.
223 [[nodiscard]] auto location() const -> Loc {
224 if (m_tok.empty())
225 return Loc{};
226 if (m_tok.size() == 1)
227 return m_tok[0].m_loc;
228
229 return m_tok[0].m_loc + m_tok[m_tok.size() - 1].m_loc;
230 };
231
232 /// A short, string representation for debugging.
233 [[nodiscard]] virtual auto describe() const -> string { return string{"unknown "} + kind_name(); }
234
235 /// Deep copy, except for the type and attachments.
236 [[nodiscard]] virtual auto clone() const -> AST* = 0;
237};
