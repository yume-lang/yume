102struct TokenIterator {
103 VectorTokenIterator m_iterator;
104 VectorTokenIterator m_end;
105
106 inline constexpr TokenIterator(const VectorTokenIterator& iterator, const VectorTokenIterator& end)
107 : m_iterator{iterator}, m_end{end} {}
108
109 [[nodiscard]] constexpr auto at_end() const noexcept -> bool { return m_iterator == m_end; }
110 [[nodiscard]] auto constexpr operator->() const noexcept -> Token* { return m_iterator.operator->(); }
111 [[nodiscard]] constexpr auto operator*() const noexcept -> Token { return m_iterator.operator*(); }
112 [[nodiscard]] constexpr auto operator+(int i) const noexcept -> TokenIterator {
113 return TokenIterator{m_iterator + i, m_end};
114 }
115 constexpr auto operator++() -> TokenIterator& {
116 if (at_end()) {
117 throw std::runtime_error("Can't increment past end");
118 }
119 ++m_iterator;
120 return *this;
121 }
122 auto operator++(int) -> TokenIterator {
123 if (at_end()) {
124 throw std::runtime_error("Can't increment past end");
125 }
126 return TokenIterator{m_iterator++, m_end};
127 }
128 [[nodiscard]] auto begin() const -> VectorTokenIterator { return m_iterator; }
129};

133struct Attachment {
134 mutable llvm::SmallPtrSet<const AST*, 2> observers{};
135 mutable llvm::SmallPtrSet<const AST*, 2> depends{};
136};

138class AST {
139protected:
140 const Kind m_kind;
141 const span<Token> m_tok;
142 mutable const ty::Type* m_val_ty{};
143 unique_ptr<Attachment> m_attach{std::make_unique<Attachment>()};
144
145 inline void unify_val_ty() const {
146 for (const auto* other : m_attach->depends) {
147 if (m_val_ty == other->m_val_ty || other->m_val_ty == nullptr) {
148 return;
149 }
150 if (m_val_ty == nullptr) {
151 m_val_ty = other->m_val_ty;
152 } else {
153 const auto* merged = m_val_ty->coalesce(*other->m_val_ty);
154 if (merged == nullptr) {
155 throw std::logic_error("Conflicting types between AST nodes that are attached: `"s + m_val_ty->name() +
156 "` vs `" + other->m_val_ty->name() + "`!");
157 }
158 m_val_ty = merged;
159 }
160 }
161 }
162
163 AST(Kind kind, span<Token> tok) : m_kind(kind), m_tok(tok) {}
164
165public:
166 AST(const AST&) = delete;
167 AST(AST&&) = default;
168 auto operator=(const AST&) -> AST& = delete;
169 auto operator=(AST&&) -> AST& = delete;
170 virtual ~AST() = default;
171
172 virtual void inline visit(Visitor& visitor) = 0;
173 [[nodiscard]] inline auto val_ty() const noexcept -> const ty::Type* { return m_val_ty; }
174 inline void val_ty(const ty::Type* type) const {
175 m_val_ty = type;
176 for (const auto* i : m_attach->observers) {
177 i->unify_val_ty();
178 }
179 }
180
181 inline void attach_to(const AST* other) const {
182 other->m_attach->observers.insert(this);
183 this->m_attach->depends.insert(other);
184 unify_val_ty();
185 }
186
187 [[nodiscard]] auto kind() const -> Kind { return m_kind; };
188 [[nodiscard]] auto kind_name() const -> string { return ast::kind_name(kind()); };
189 [[nodiscard]] auto token_range() const -> const span<Token>& { return m_tok; };
190 [[nodiscard]] auto location() const -> Loc {
191 if (m_tok.empty()) {
192 return Loc{};
193 }
194 if (m_tok.size() == 1) {
195 return m_tok[0].m_loc;
196 }
197 return m_tok[0].m_loc + m_tok[m_tok.size() - 1].m_loc;
198 };
199 [[nodiscard]] virtual auto inline describe() const -> string { return string{"unknown "} + kind_name(); }
200 [[nodiscard]] virtual auto clone() const -> AST* = 0;
201};
