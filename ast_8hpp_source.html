128/// Represents "any" kind of ast node of type `T`, or the absence of one. See `OptionalExpr` and `OptionalType`.
129/**
130 * This template exists to avoid passing around `unique_ptr<T>`s around in code that deals with AST nodes containing
131 * other generic nodes, such as `Compound`.
132 * This class has the exact same layout at `AnyBase`, but is nominally explicit in its semantics, in that it may be
133 * null. It replaces the earlier usages of `optional<unique_ptr<T>>` which in theory wastes memory for the "optional"
134 * part when a nullptr already expresses the desired semantics.
135 * Unlike `AnyBase`, this class does not perform null pointer checks, and may be default constructed, or constructed
136 * from a `nullptr`.
137 */
138template <typename T> class OptionalAnyBase {
139 unique_ptr<T> m_val;
140 friend AnyBase<T>;
141
142public:
143 OptionalAnyBase() : m_val() {}
144 explicit OptionalAnyBase(T* raw_ptr) : m_val{raw_ptr} {}
145 template <std::convertible_to<unique_ptr<T>> U> OptionalAnyBase(U uptr) : m_val{move(uptr)} {}
146 OptionalAnyBase(std::nullopt_t /* tag */) : m_val{} {}
147 explicit OptionalAnyBase(optional<unique_ptr<T>> opt_uptr)
148 : m_val{opt_uptr.has_value() ? move(*opt_uptr) : nullptr} {}
149
150 [[nodiscard]] auto operator->() const -> const T* { return m_val.operator->(); }
151 [[nodiscard]] auto operator*() const -> const T& { return *m_val; }
152 [[nodiscard]] auto operator->() -> T* { return m_val.operator->(); }
153 [[nodiscard]] auto operator*() -> T& { return *m_val; }
154
155 [[nodiscard]] operator bool() const { return static_cast<bool>(m_val); }
156 [[nodiscard]] auto has_value() const -> bool { return static_cast<bool>(m_val); }
157 [[nodiscard]] auto raw_ptr() const -> const T* { return m_val.get(); }
158 [[nodiscard]] auto raw_ptr() -> T* { return m_val.get(); }
159};

200/// All nodes in the `AST` tree of the program inherit from this class.
201/**
202 * AST nodes cannot be copied or moved as all special member functions (aside from the destructor) are `delete`d. There
203 * will always be one instance for a specific node during the lifetype of the compiler.
204 * However note that AST nodes can be cloned with the `clone()` method, which will produce a deep copy. This is used
205 * when instantiating a template. The clone will have the exact same structure and locations but distinct types,
206 * depending on what the types the template is instantiated with.
207 */
208class AST {
209 /// Every subclass of `AST` has a distinct `Kind`.
210 const Kind m_kind;
211 /// The range of tokenizer `Token`s that this node was parsed from.
212 const span<Token> m_tok;
213 /// The value type of this node. Determined in the semantic phase; always empty after parsing.
214 optional<ty::Type> m_val_ty{};
215 /// \see Attachment
216 unique_ptr<Attachment> m_attach{std::make_unique<Attachment>()};
217
218protected:
219 /// Verify the type compatibility of the depends of this node, and merge the types if possible.
220 /// This is called every time the node's type is updated.
221 void unify_val_ty();
222
223 [[nodiscard]] auto tok() const noexcept -> span<Token> { return m_tok; }
224
225 AST(Kind kind, span<Token> tok) : m_kind(kind), m_tok(tok) {}
226
227public:
228 AST(const AST&) = delete;
229 AST(AST&&) = default;
230 auto operator=(const AST&) -> AST& = delete;
231 auto operator=(AST&&) -> AST& = delete;
232 virtual ~AST() = default;
233
234 /// Recursively visit this ast node and all its constituents. \see Visitor
235 virtual void visit(Visitor& visitor) const = 0;
236
237 [[nodiscard]] auto val_ty() const noexcept -> optional<ty::Type> { return m_val_ty; }
238 [[nodiscard]] auto ensure_ty() const -> ty::Type {
239 yume_assert(m_val_ty.has_value(), "Ensured that AST node has type, but one has not been assigned");
240 // NOLINTNEXTLINE(bugprone-unchecked-optional-access): clang-tidy doesn't accept yume_assert as an assertion
241 return *m_val_ty;
242 }
243 void val_ty(optional<ty::Type> type) {
244 m_val_ty = type;
245 for (auto* i : m_attach->observers) {
246 i->unify_val_ty();
247 }
248 }
249
250 /// Make the type of this node depend on the type of `other`.
251 /// \sa Attachment
252 void attach_to(nonnull<AST*> other) {
253 other->m_attach->observers.insert(this);
254 this->m_attach->depends.insert(other);
255 unify_val_ty();
256 }
257
258 [[nodiscard]] auto kind() const -> Kind { return m_kind; };
259 /// Human-readable string representation of the `Kind` of this node.
260 [[nodiscard]] auto kind_name() const -> string { return ast::kind_name(kind()); };
261 [[nodiscard]] auto token_range() const -> const span<Token>& { return m_tok; };
262
263 /// The union of the locations of the `Token`s making up this node.
264 [[nodiscard]] auto location() const -> Loc;
265
266 /// A short, string representation for debugging.
267 [[nodiscard]] virtual auto describe() const -> string;
268
269 /// Deep copy, except for the type and attachments.
270 [[nodiscard]] virtual auto clone() const -> AST* = 0;
271};
