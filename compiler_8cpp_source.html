130void Compiler::declare_default_ctor(Struct& st) {
131 if (st.ast().is_interface)
132 return; // Don't declare implicit ctors if the struct is an interface
133
134 const bool no_ctors_declared =
135 std::ranges::none_of(m_ctors, [&](const Fn& fn) { return fn.get_self_ty() == st.get_self_ty(); });
136
137 if (!no_ctors_declared)
138 return; // Don't declare implicit ctors if at least one user-defined one exists
139
140 vector<ast::TypeName> ctor_args;
141 vector<ast::AnyStmt> ctor_body;
142 for (auto& field : st.ast().fields) {
143 auto tok = field.token_range();
144 ast::AnyType proxy_type = make_unique<ast::ProxyType>(tok, field.name);
145 ctor_args.emplace_back(field.token_range(), move(proxy_type), field.name);
146
147 auto implicit_field = make_unique<ast::FieldAccessExpr>(tok, std::nullopt, field.name);
148 auto arg_var = make_unique<ast::VarExpr>(tok, field.name);
149 ctor_body.emplace_back(make_unique<ast::AssignExpr>(tok, move(implicit_field), move(arg_var)));
150 }
151 // TODO(rymiel): Give these things sensible locations?
152 auto& new_ct = st.body().body.emplace_back(
153 std::make_unique<ast::CtorDecl>(span<Token>{}, move(ctor_args), ast::Compound({}, move(ctor_body))));
154
155 walk_types(decl_statement(*new_ct, st.get_self_ty(), st.member));
156}

177void Compiler::run() {
178 m_scope.push_scope(); // Global scope
179
180 for (const auto& source : m_sources)
181 for (auto& i : source.program->body)
182 decl_statement(*i, {}, source.program.get());
183
184 // 1: Only convert the types of constants
185 for (auto& cn : m_consts)
186 walk_types(&cn);
187
188 // 2: Only convert structs
189 for (auto& st : m_structs)
190 walk_types(&st);
191
192 // 3: Convert initializers of constants
193 for (auto& cn : m_consts) {
194 auto* const_ty = llvm_type(cn.ast().ensure_ty());
195 cn.llvm = new llvm::GlobalVariable(*m_module, const_ty, false, llvm::GlobalVariable::PrivateLinkage, nullptr,
196 ".const." + cn.name());
197 }
198
199 // 4: Only convert user defined constructors
200 for (auto& ct : m_ctors)
201 walk_types(&ct);
202
203 // At this point, all _user defined_ constructors have been declared, so we can add implicitly defined constructors to
204 // structs which haven't declared any.
205 for (auto& st : m_structs)
206 declare_default_ctor(st);
207
208 // 5: only convert function parameters
209 for (auto& fn : m_fns)
210 walk_types(&fn);
211
212 // 6: Create vtables for interfaces
213 for (auto& st : m_structs)
214 if (st.ast().is_interface)
215 create_vtable_for(st);
216
217 // 7: convert everything else, but only when instantiated
218 m_walker->in_depth = true;
219
220 for (auto& cn : m_consts) {
221 walk_types(&cn);
222 define(cn);
223 }
224
225 // Find all external functions. These will be the "entrypoints".
226 vector<Fn*> extern_fns = {};
227 for (auto& fn : m_fns) {
228 if (fn.name() == "main")
229 fn.make_extern_linkage();
230
231 if (fn.extern_linkage() && !fn.extern_decl())
232 extern_fns.push_back(&fn);
233 }
234
235 if (extern_fns.empty()) {
236 throw std::logic_error("Program is missing any declarations with external linkage. Perhaps you meant to declare a "
237 "`main` function?"); // Related: #10
238 }
239 for (auto* ext : extern_fns)
240 declare(*ext);
241
242 while (!m_decl_queue.empty()) {
243 auto next = m_decl_queue.front();
244 m_decl_queue.pop();
245 next.visit([](std::monostate /*absent*/) { /* nothing to do */ }, //
246 [&](Fn* fn) { define(*fn); }, //
247 [&](Const* cn) { define(*cn); },
248 [&](Struct* /*st*/) { throw std::logic_error("Cannot define a struct"); });
249 }
250
251 yume_assert(m_scope.size() == 1, "End of compilation should end with only the global scope remaining");
252
253 m_builder->SetInsertPoint(&m_global_dtor_fn->getEntryBlock(), m_global_dtor_fn->getEntryBlock().begin());
254 destruct_last_scope();
255 m_scope.clear();
256
257 m_debug->finalize();
258
259 if (llvm::verifyModule(*m_module, &errs())) {
260 m_module->print(errs(), nullptr, false, true);
261 throw std::runtime_error("Module verification failed");
262 }
263}

274auto Compiler::create_struct(Struct& st) -> bool {
275 auto& s_decl = st.st_ast;
276
277 auto fields = vector<ast::TypeName*>();
278 fields.reserve(s_decl.fields.size());
279 for (auto& f : s_decl.fields)
280 fields.push_back(&f);
281
282 auto iter = m_types.known.find(s_decl.name);
283 if (iter == m_types.known.end()) {
284 auto empl =
285 m_types.known.try_emplace(s_decl.name, std::make_unique<ty::Struct>(s_decl.name, move(fields), &st, &st.subs));
286 yume_assert(isa<ty::Struct>(*empl.first->second));
287 st.self_ty = &*empl.first->second;
288 return true;
289 }
290
291 yume_assert(isa<ty::Struct>(*iter->second));
292 auto& existing = cast<ty::Struct>(*iter->second);
293

295 return false;
296
297 existing.m_fields = move(fields);
298 st.self_ty = &existing.emplace_subbed(st.subs);
299 return true;

302auto Compiler::decl_statement(ast::Stmt& stmt, optional<ty::Type> parent, ast::Program* member) -> DeclLike {
303 if (auto* fn_decl = dyn_cast<ast::FnDecl>(&stmt)) {
304 vector<unique_ptr<ty::Generic>> type_args{};
305 Substitution subs{};
306 for (auto& i : fn_decl->type_args) {
307 auto& gen = type_args.emplace_back(std::make_unique<ty::Generic>(i));
308 subs.try_emplace(i, gen.get());
309 }
310 auto& fn = m_fns.emplace_back(fn_decl, member, parent, move(subs), move(type_args));
311 fn_decl->sema_decl = &fn;
312
313 return &fn;
314 }
315 if (auto* s_decl = dyn_cast<ast::StructDecl>(&stmt)) {
316 vector<unique_ptr<ty::Generic>> type_args{};
317 Substitution subs{};
318 for (auto& i : s_decl->type_args) {
319 auto& gen = type_args.emplace_back(std::make_unique<ty::Generic>(i));
320 subs.try_emplace(i, gen.get());
321 }
322 auto& st = m_structs.emplace_back(*s_decl, member, std::nullopt, subs, move(type_args));
323 if (!create_struct(st)) {
324 m_structs.pop_back();
325 return {};
326 }
327
328 if (st.name() == "Slice") // TODO(rymiel): magic value?
329 m_slice_struct = &st;
330
331 for (auto& f : s_decl->body)
332 if (st.type_args.empty() || isa<ast::CtorDecl>(*f))
333 decl_statement(*f, st.self_ty, member);
334
335 return &st;
336 }
337 if (auto* ctor_decl = dyn_cast<ast::CtorDecl>(&stmt)) {
338 auto& ctor = m_ctors.emplace_back(ctor_decl, member, parent);
339
340 return &ctor;
341 }
342 if (auto* const_decl = dyn_cast<ast::ConstDecl>(&stmt)) {
343 auto& cn = m_consts.emplace_back(*const_decl, member, parent);
344
345 return &cn;
346 }
347
348 throw std::runtime_error("Invalid top-level statement: "s + stmt.kind_name());
349}

351static auto build_function_type(Compiler& compiler, const ty::Function& type) -> llvm::Type* {
352 llvm::StructType* closure_ty = nullptr;
353 llvm::Type* memo = nullptr;
354 auto* erased_closure_ty = compiler.builder()->getInt8PtrTy();
355
356 if (!type.is_fn_ptr()) {
357 auto closured = vector<llvm::Type*>{};
358 for (const auto& i : type.closure())
359 closured.push_back(compiler.llvm_type(i));
360 if (closured.empty()) // Can't have an empty closure type
361 closured.push_back(compiler.builder()->getInt8Ty());
362
363 // Closures are type-erased to just a bag of bits. We store the "real type" of the closure within the function
364 // type for better retrieval, but in practice, they're always passed around bitcasted. Note that the bitcasting
365 // is not required when opaque pointers are in play
366 closure_ty = llvm::StructType::create(closured, "closure");
367 }
368
369 auto args = vector<llvm::Type*>{};
370 if (closure_ty != nullptr)
371 args.push_back(erased_closure_ty); // Lambda takes a closure as the first parameter
372 for (const auto& i : type.args())
373 args.push_back(compiler.llvm_type(i));
374
375 auto* return_type = compiler.builder()->getVoidTy();
376 if (auto ret = type.ret(); ret.has_value())
377 return_type = compiler.llvm_type(*ret);
378
379 auto* fn_ty = llvm::FunctionType::get(return_type, args, type.is_c_varargs());
380
381 if (type.is_fn_ptr())
382 memo = fn_ty->getPointerTo();
383 else
384 memo = llvm::StructType::get(fn_ty->getPointerTo(), erased_closure_ty);
385
386 type.fn_memo(compiler, fn_ty);
387 type.closure_memo(compiler, closure_ty);
388 type.memo(compiler, memo);
389 return memo;
390}

407auto Compiler::llvm_type(ty::Type type, bool erase_opaque) -> llvm::Type* {
408 llvm::Type* base = nullptr;
409 bool interface_self = false;
410
411 if (const auto* int_type = type.base_dyn_cast<ty::Int>()) {
412 base = llvm::Type::getIntNTy(*m_context, int_type->size());
413 } else if (const auto* ptr_type = type.base_dyn_cast<ty::Ptr>()) {
414 yume_assert(ptr_type->qualifier() == Qualifier::Ptr, "Ptr type must hold pointer");
415 base = llvm::PointerType::getUnqual(llvm_type(ptr_type->pointee()));
416 } else if (const auto* struct_type = type.base_dyn_cast<ty::Struct>()) {
417 llvm::Type* memo = struct_type->memo();
418 if (memo == nullptr)
419 memo = build_struct_type(*this, *struct_type);
420
421 base = memo;
422 interface_self = struct_type->is_interface();
423 } else if (const auto* function_type = type.base_dyn_cast<ty::Function>()) {
424 llvm::Type* memo = function_type->memo();
425 if (memo == nullptr)
426 memo = build_function_type(*this, *function_type);
427
428 base = memo;
429 } else if (type.base_isa<ty::Nil>()) {
430 base = llvm::Type::getVoidTy(*m_context);
431 } else if (const auto* opaque_self_type = type.base_dyn_cast<ty::OpaqueSelf>()) {
432 if (erase_opaque)
433 return m_builder->getInt8PtrTy();
434 base = llvm_type({opaque_self_type->indirect()});
435 interface_self = llvm::isa<ty::Struct>(opaque_self_type->indirect()) &&
436 llvm::cast<ty::Struct>(opaque_self_type->indirect())->is_interface();
437 } else if (type.base_isa<ty::Meta>()) {
438 base = m_builder->getInt8Ty();
439 } else {
440 throw std::logic_error("Unknown type base " + type.name());
441 }
442
443 if (type.is_mut() && !interface_self)
444 return base->getPointerTo();
445 if (type.is_opaque_self() && !interface_self)
446 return base->getPointerTo();
447 return base;
448}

486 }

496auto Compiler::default_init(ty::Type type) -> Val {
497 if (type.is_mut())
498 throw std::runtime_error("Cannot default-initialize a reference");
499 if (const auto* int_type = type.base_dyn_cast<ty::Int>())
500 return m_builder->getIntN(int_type->size(), 0);
501 if (const auto* ptr_type = type.base_dyn_cast<ty::Ptr>()) {
502 switch (ptr_type->qualifier()) {
503 default: llvm_unreachable("Ptr type cannot hold this qualifier");
504 case Qualifier::Ptr:
505 llvm::ConstantPointerNull::get(llvm::PointerType::getUnqual(llvm_type(ptr_type->pointee())));
506 break;
507 }
508 }
509 if (const auto* struct_type = type.base_dyn_cast<ty::Struct>()) {
510 auto* llvm_ty = cast<llvm::StructType>(llvm_type(type));
511 Val val = llvm::UndefValue::get(llvm_ty);
512
513 for (const auto& i : llvm::enumerate(struct_type->fields()))
514 val = m_builder->CreateInsertValue(val, default_init(i.value()->type->ensure_ty()), i.index());

522auto Compiler::declare(Fn& fn) -> llvm::Function* {
523 if (fn.llvm != nullptr)
524 return fn.llvm;
525 if (fn.primitive())
526 return nullptr;
527
528 yume_assert(fn.fn_ty != nullptr, "Function declaration "s + fn.name() + " has no function type set");
529 (void)llvm_type(fn.fn_ty); // Ensure we've created a function type
530 llvm::FunctionType* fn_t = fn.fn_ty->fn_memo();
531
532 string name = fn.name();
533 if (!fn.extern_linkage() && !fn.local())
534 name = mangle::mangle_name(fn);
535
536 auto linkage = fn.extern_linkage() ? llvm::Function::ExternalLinkage
537 : fn.local() ? llvm::Function::PrivateLinkage
538 : llvm::Function::InternalLinkage;
539 auto* llvm_fn = llvm::Function::Create(fn_t, linkage, name, m_module.get());
540 if (fn.has_annotation("interrupt") && fn.fn_ty->closure_memo() == nullptr) // HACK
541 llvm_fn->setCallingConv(llvm::CallingConv::X86_INTR);
542 fn.llvm = llvm_fn;
543
544 auto arg_names = fn.arg_names();
545 if (fn.fn_ty->closure_memo() != nullptr)
546 arg_names.insert(arg_names.begin(), "<closure>"); // For functions with closures, it is the first argument
547 for (auto [llvm_arg, arg_name] : llvm::zip(llvm_fn->args(), arg_names))
548 llvm_arg.setName("arg."s + arg_name);
549
550 // At this point, the function prototype is declared, but not the body.
551 // In the case of extern or local functions, a prototype is all that will be declared.
552 if (!fn.extern_decl() && !fn.local()) {
553 auto* fn_unit = m_source_mapping.at(fn.member);
554 auto* fn_file = m_debug->createFile(fn_unit->getFilename(), fn_unit->getDirectory());
555 llvm::DIScope* f_context = fn_file;
556 unsigned line_no = fn.ast().location().begin_line;
557 unsigned scope_line = line_no;
558 auto types = m_debug->getOrCreateTypeArray({}); // TODO(rymiel)
559 llvm::DISubprogram* subprogram =
560 m_debug->createFunction(f_context, fn.name(), name, fn_file, line_no, m_debug->createSubroutineType(types),
561 scope_line, llvm::DINode::FlagPrototyped, llvm::DISubprogram::SPFlagDefinition);
562 fn.llvm->setSubprogram(subprogram);
563
564 m_decl_queue.emplace(&fn);
565 m_walker->current_decl = &fn;
566 m_walker->body_statement(fn.ast());
567 }
568 return llvm_fn;
569}

612void Compiler::setup_fn_base(Fn& fn) {
613 m_current_fn = &fn;
614 m_scope.push_scope();
615 m_scope_ctor.reset();

618 m_builder->SetCurrentDebugLocation({});

626 // Allocate local variables for each parameter

631void Compiler::define(Const& cn) {
632 if (cn.llvm->hasInitializer())
633 return;
634
635 auto* saved_insert_block = m_builder->GetInsertBlock();
636 m_builder->SetInsertPoint(&m_global_ctor_fn->getEntryBlock(), m_global_ctor_fn->getEntryBlock().begin());
637
638 auto init = body_expression(*cn.ast().init);
639 if ((init.scope != nullptr) && init.scope->owning)
640 init.scope->owning = false;
641
642 if (isa<llvm::Constant>(init.llvm)) {
643 cn.llvm->setConstant(true);
644 auto* const_val = cast<llvm::Constant>(init.llvm);
645 cn.llvm->setInitializer(const_val);
646 } else {
647 cn.llvm->setInitializer(llvm::ConstantAggregateZero::get(cn.llvm->getValueType()));
648 m_builder->CreateStore(init.llvm, cn.llvm);
649 }
650
651 auto cn_type = cn.ast().type->ensure_ty();
652 if (!cn_type.is_trivially_destructible()) {
653 m_builder->SetInsertPoint(&m_global_dtor_fn->getEntryBlock(), m_global_dtor_fn->getEntryBlock().begin());
654 destruct(m_builder->CreateLoad(llvm_type(cn_type), cn.llvm), cn_type);
655 }
656
657 m_builder->SetInsertPoint(saved_insert_block);
658}

660void Compiler::define(Fn& fn) {
661 setup_fn_base(fn);
662
663 if (fn.abstract()) {
664 yume_assert(fn.self_ty.has_value(), "Abstract function must refer to a self type");
665 yume_assert(fn.self_ty->base_isa<ty::Struct>(), "Abstract function must be within a struct");
666 const auto* interface = fn.self_ty->base_cast<ty::Struct>()->decl();
667 yume_assert(interface != nullptr, "Struct not found from struct type?");
668 yume_assert(interface->ast().is_interface, "Abstract function must be within interface");
669
670 auto vtable_match = std::ranges::find(interface->vtable_members, vtable_entry_for(fn));
671 yume_assert(vtable_match != interface->vtable_members.end(), "abstract method not found in vtable?");
672 auto vtable_entry_index = std::distance(interface->vtable_members.begin(), vtable_match);
673
674 auto vtable_deref_args = vector<llvm::Type*>();
675 auto* vtable_deref_ret = m_builder->getVoidTy();
676 for (const auto& arg : vtable_match->args)
677 vtable_deref_args.emplace_back(llvm_type(arg, true));
678 if (vtable_match->ret)
679 vtable_deref_ret = llvm_type(*vtable_match->ret, true);
680
681 auto call_args = vector<llvm::Value*>();
682 llvm::FunctionType* call_fn_type = nullptr;
683
684 // An empty arg name means it is actually fake and not in impl methods. this is a HACK
685 if (fn.arg_count() == 1 && fn.arg_names().at(0).empty()) {
686 call_fn_type = llvm::FunctionType::get(vtable_deref_ret, {}, false);
687 } else {
688 call_args.push_back(m_builder->CreateExtractValue(fn.llvm->args().begin(), 1, "abs.self"));
689 for (auto& extra_arg : llvm::drop_begin(fn.llvm->args()))
690 call_args.emplace_back(&extra_arg);
691
692 call_fn_type = llvm::FunctionType::get(vtable_deref_ret, vtable_deref_args, false);
693 }
694
695 // TODO(LLVM MIN >= 15): A lot of bit-twiddling obsoleted by opaque pointers
696
697 auto* vtable_struct_ptr = m_builder->CreateExtractValue(fn.llvm->args().begin(), 0, "abs.vt.struct");
698 auto* vtable_ptr_array =
699 m_builder->CreateBitCast(vtable_struct_ptr, m_builder->getInt8PtrTy()->getPointerTo(), "abs.vt.array");
700 auto* vtable_entry = m_builder->CreateLoad(
701 m_builder->getInt8PtrTy(),
702 m_builder->CreateConstGEP1_32(m_builder->getInt8PtrTy(), vtable_ptr_array, vtable_entry_index, "abs.vt.entry"),
703 "abs.vt.entry.load");
704 Val call_result = m_builder->CreateCall(
705 call_fn_type, m_builder->CreateBitCast(vtable_entry, call_fn_type->getPointerTo(), "abs.fn"), call_args);
706
707 if (call_result.llvm->getType()->isVoidTy())
708 m_builder->CreateRetVoid();
709 else
710 m_builder->CreateRet(call_result);
711 } else if (isa<ast::FnDecl>(&fn.ast())) {
712 if (auto* body = get_if<ast::Compound>(&fn.fn_body()); body != nullptr)
713 statement(*body);
714
715 if (m_builder->GetInsertBlock()->getTerminator() == nullptr) {
716 destruct_all_scopes();
717 m_builder->CreateRetVoid();
718 }
719 } else {
720 yume_assert(fn.self_ty.has_value(), "Cannot define constructor when the type being constructed is unknown");
721 // NOLINTNEXTLINE(bugprone-unchecked-optional-access): clang-tidy doesn't accept yume_assert as an assertion
722 auto* ctor_type = llvm_type(*fn.self_ty);
723 const Val base_value = llvm::UndefValue::get(ctor_type);
724 auto* base_alloc = m_builder->CreateAlloca(ctor_type, nullptr, "ctor.base");
725 m_builder->CreateStore(base_value, base_alloc);
726
727 // Act as if we don't own the object being constructed so it won't get destructed at the end of scope
728 m_scope_ctor.emplace(InScope{.value = base_alloc, .ast = fn.ast(), .owning = false});
729
730 statement(fn.compound_body());
731
732 destruct_all_scopes();
733 auto* finalized_value = m_builder->CreateLoad(ctor_type, base_alloc);
734 m_builder->CreateRet(finalized_value);
735 }
736
737 m_scope.pop_scope();
738 yume_assert(m_scope.size() == 1, "End of function should end with only the global scope remaining");
739 // llvm::verifyFunction(*fn.llvm, &errs());
740}

742template <> void Compiler::statement(ast::WhileStmt& stat) {
743 auto* test_bb = llvm::BasicBlock::Create(*m_context, "while.test", m_current_fn->llvm);
744 auto* head_bb = llvm::BasicBlock::Create(*m_context, "while.head", m_current_fn->llvm);
745 auto* merge_bb = llvm::BasicBlock::Create(*m_context, "while.merge", m_current_fn->llvm);
746 m_builder->CreateBr(test_bb);
747 m_builder->SetInsertPoint(test_bb);
748 auto cond_value = body_expression(*stat.cond);
749 m_builder->CreateCondBr(cond_value, head_bb, merge_bb);
750 m_builder->SetInsertPoint(head_bb);
751 body_statement(stat.body);
752 m_builder->CreateBr(test_bb);
753 m_builder->SetInsertPoint(merge_bb);
754}

756template <> void Compiler::statement(ast::IfStmt& stat) {
757 auto* merge_bb = llvm::BasicBlock::Create(*m_context, "if.cont", m_current_fn->llvm);
758 auto* next_test_bb = llvm::BasicBlock::Create(*m_context, "if.test", m_current_fn->llvm, merge_bb);
759 auto* last_branch = m_builder->CreateBr(next_test_bb);
760 bool all_terminated = true;
761
762 auto& clauses = stat.clauses;
763 for (auto& clause : clauses) {
764 m_builder->SetInsertPoint(next_test_bb);
765 auto* body_bb = llvm::BasicBlock::Create(*m_context, "if.then", m_current_fn->llvm, merge_bb);
766 next_test_bb = llvm::BasicBlock::Create(*m_context, "if.test", m_current_fn->llvm, merge_bb);
767 auto condition = body_expression(*clause.cond);
768 last_branch = m_builder->CreateCondBr(condition, body_bb, next_test_bb);
769 m_builder->SetInsertPoint(body_bb);
770 statement(clause.body);
771 // This has to use GetInsertBlock() since the statement being created above can introduce many new blocks.
772 if (m_builder->GetInsertBlock()->getTerminator() == nullptr) {
773 all_terminated = false;
774 m_builder->CreateBr(merge_bb);
775 }
776 }
777
778 auto& else_clause = stat.else_clause;
779 if (else_clause.has_value()) {
780 next_test_bb->setName("if.else");
781 m_builder->SetInsertPoint(next_test_bb);
782 statement(*else_clause);
783 if (m_builder->GetInsertBlock()->getTerminator() == nullptr) {
784 all_terminated = false;
785 m_builder->CreateBr(merge_bb);
786 }
787 } else {
788 // We can't consider all branches to terminate when there is no else clause, as the whole if statement could be
789 // skipped
790 all_terminated = false;
791 last_branch->setSuccessor(1, merge_bb);
792 next_test_bb->eraseFromParent();
793 }
794
795 if (all_terminated)
796 merge_bb->eraseFromParent();
797 else
798 m_builder->SetInsertPoint(merge_bb);
799}

801template <> void Compiler::statement(ast::ReturnStmt& stat) {
802 InScope* reset_owning = nullptr;
803
804 if (stat.expr.has_value()) {
805 // Returning a local variable also gives up ownership of it
806 if (stat.extends_lifetime != nullptr) {
807 for (auto& i : m_scope.last_scope()) {
808 auto& v = i.second;
809 if (&v.ast == stat.extends_lifetime) {
810 v.owning = false;
811 reset_owning = &v;
812 break;
813 }
814 }
815 }
816
817 m_return_value = stat.expr.raw_ptr(); // TODO(rymiel): Find a better solution than this global variable
818 auto val = body_expression(*stat.expr);
819
820 if (val.scope != nullptr && val.scope->owning)
821 val.scope->owning = false;
822
823 destruct_all_scopes();
824
825 if (reset_owning != nullptr)
826 reset_owning->owning = true; // The local variable may not be returned in all code paths, so reset its ownership
827
828 if (val.llvm->getType()->isVoidTy() || m_current_fn->llvm->getReturnType()->isVoidTy())
829 m_builder->CreateRetVoid();
830 else
831 m_builder->CreateRet(val);
832
833 return;
834 }
835
836 destruct_all_scopes();
837 m_builder->CreateRetVoid();
838}

966 const auto& a = args.at(0);
967 const auto& b = args.at(1);
968 auto hash = const_hash(primitive.data());
969 switch (hash) {
970 case const_hash("ib_icmp_sgt"): return m_builder->CreateICmpSGT(a, b);

973 case const_hash("ib_icmp_ult"): return m_builder->CreateICmpULT(a, b);
974 case const_hash("ib_icmp_eq"): return m_builder->CreateICmpEQ(a, b);
975 case const_hash("ib_icmp_ne"): return m_builder->CreateICmpNE(a, b);
976 case const_hash("ib_add"): return m_builder->CreateAdd(a, b);
977 case const_hash("ib_sub"): return m_builder->CreateSub(a, b);
978 case const_hash("ib_mul"): return m_builder->CreateMul(a, b);
979 case const_hash("ib_and"): return m_builder->CreateAnd(a, b);
980 case const_hash("ib_srem"): return m_builder->CreateSRem(a, b);
981 case const_hash("ib_urem"): return m_builder->CreateURem(a, b);
982 case const_hash("ib_sdiv"): return m_builder->CreateSDiv(a, b);
983 case const_hash("ib_udiv"): return m_builder->CreateUDiv(a, b);
984 case const_hash("ib_shl"): return m_builder->CreateShl(a, b);
985 case const_hash("ib_lshr"): return m_builder->CreateLShr(a, b);
986 case const_hash("ib_ashr"): return m_builder->CreateAShr(a, b);
987 default: throw std::runtime_error("Unknown binary integer primitive "s + primitive);
988 }

999auto Compiler::primitive(Fn* fn, const vector<Val>& args, const vector<ty::Type>& types,
1000 vector<ast::AnyExpr*>& ast_args) -> optional<Val> {
1001 if (fn->extern_decl())
1002 return m_builder->CreateCall(declare(*fn), vals_to_llvm(args));
1003
1004 if (!fn->primitive())
1005 return {};
1006
1007 auto primitive = get<string>(fn->fn_body());
1008
1009 if (primitive == "ptrto")
1010 return args.at(0);
1011 if (primitive == "slice_malloc") {
1012 auto base_ty_type = types.at(0).ensure_ptr_base();
1013 auto* base_type = llvm_type(base_ty_type);
1014 auto slice_size = args.at(1);
1015
1016 return create_malloc(base_type, slice_size, "sl.ctor.malloc");
1017 }
1018 if (primitive == "default_init") {
1019 auto base_type = types.at(0).ensure_mut_base();
1020 m_builder->CreateStore(default_init(base_type), args.at(0));
1021
1022 return args.at(0);
1023 }
1024 if (primitive == "set_at") {
1025 auto* result_type = llvm_type(types[0].without_mut().ensure_ptr_base());
1026 llvm::Value* value = args.at(2);
1027 llvm::Value* base = m_builder->CreateGEP(result_type, args.at(0).llvm, args.at(1).llvm, "p.set_at.gep");
1028 m_builder->CreateStore(value, base);
1029 return llvm::UndefValue::get(m_builder->getVoidTy());
1030 }
1031 if (primitive == "get_at") {
1032 auto* result_type = llvm_type(types[0].without_mut().ensure_ptr_base());
1033 llvm::Value* base = args.at(0);
1034 base = m_builder->CreateGEP(result_type, base, args.at(1).llvm, "p.get_at.gep");
1035 return base;
1036 }
1037 if (primitive == "ptr_cast")
1038 return m_builder->CreateBitCast(args[0], llvm_type(types.at(1).without_meta()), "builtin.ptr_cast");
1039 if (primitive == "ptr_gep") {
1040 return m_builder->CreateGEP(llvm_type(types.at(0).ensure_ptr_base()), args.at(0), args.at(1).llvm,
1041 "builtin.ptr_gep");
1042 }
1043 if (primitive == "cast") {
1044 // TODO(rymiel): This is an "explicit" cast, and should be able to cast more things when compared to an implicit one
1045 auto* base = ast_args.at(0);
1046 semantic::make_implicit_conversion(*base, types.at(1).without_meta());
1047 return body_expression(**base);
1048 }
1049 if (primitive.starts_with("ib_"))
1050 return int_bin_primitive(primitive, args);
1051 throw std::runtime_error("Unknown primitive "s + primitive);
1052}

1054template <> auto Compiler::expression(ast::CallExpr& expr) -> Val {
1055 if (expr.name == "->") // TODO(rymiel): Magic value?
1056 return direct_call_operator(expr);
1057
1058 auto* selected = expr.selected_overload;
1059 llvm::Function* llvm_fn = nullptr;
1060
1061 vector<ast::AnyExpr*> ast_args{};
1062 vector<ty::Type> arg_types{};
1063 vector<Val> llvm_args{};
1064
1065 for (auto& i : expr.args) {

1088template <> auto Compiler::expression(ast::BinaryLogicExpr& expr) -> Val {
1089 bool is_and = expr.operation == "&&"_a;
1090 string label_base = (is_and ? "land" : "lor");
1091 auto* entry_block = m_builder->GetInsertBlock();

1093 auto* pass_block = llvm::BasicBlock::Create(*m_context, label_base + ".pass", m_current_fn->llvm);
1094 auto lhs_val = body_expression(*expr.lhs);
1095 if (is_and)
1096 m_builder->CreateCondBr(lhs_val, rhs_block, pass_block);
1097 else
1098 m_builder->CreateCondBr(lhs_val, pass_block, rhs_block);
1099 m_builder->SetInsertPoint(rhs_block);
1100 auto rhs_val = body_expression(*expr.rhs);
1101 m_builder->CreateBr(pass_block);
1102 m_builder->SetInsertPoint(pass_block);
1103 auto* phi = m_builder->CreatePHI(llvm_type(expr.ensure_ty()), 2, label_base + ".phi");
1104 if (is_and)
1105 phi->addIncoming(m_builder->getFalse(), entry_block);
1106 else
1107 phi->addIncoming(m_builder->getTrue(), entry_block);
1108 phi->addIncoming(rhs_val, rhs_block);
1109 return phi;
