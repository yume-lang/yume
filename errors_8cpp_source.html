93auto stacktrace_ostream::simplify(string_view msg, ContainerLikeSimplify s) -> bool {
94 constexpr static const string_view MIDDLE_END = "> >";
95 constexpr static const string_view END = ">";
96
97 auto uptr_start = msg.find(s.front);
98 if (uptr_start == string::npos)
99 return false;
100
101 auto uptr_delete_start = msg.find(s.middle, uptr_start);
102 if (uptr_delete_start == string::npos)
103 return false;
104
105 auto contained_start = uptr_start + s.front.length();
106 auto contained_typename = msg.substr(contained_start, uptr_delete_start - contained_start);
107 auto from_deleter = msg.substr(uptr_delete_start + s.middle.length());
108
109 if (!from_deleter.starts_with(contained_typename))
110 return false;
111
112 if (!from_deleter.substr(contained_typename.length()).starts_with(MIDDLE_END))
113 return false;
114
115 auto uptr_end = uptr_delete_start + s.middle.length() + contained_typename.length() + MIDDLE_END.length();
116
117 simplify(msg.substr(0, uptr_start));
118 m_buffer << s.front << contained_typename << END;
119 simplify(msg.substr(uptr_end));
120
121 return true;
122};

136void stacktrace_ostream::simplify(string_view msg) {
137 bool found_any = simplify(msg, UPTR_SIMPLIFY) || simplify(msg, VEC_SIMPLIFY) || simplify(msg, STRING_SIMPLIFY) ||
138 simplify(msg, STRINGSTREAM_SIMPLIFY) || simplify(msg, ANONYMOUS_NS_SIMPLIFY);
139
140 if (found_any)
141 return;
142
143 size_t prev_pos = 0;
144 size_t start_pos = 0;
145 while (true) {
146 start_pos = msg.find_first_of("<>()", prev_pos);
147 auto substr = msg.substr(prev_pos, start_pos == string::npos ? start_pos : start_pos - prev_pos);
148
149 if (m_template_depth == 0)
150 set_color(CYAN);
151
152 m_buffer << substr;
153
154 if (m_template_depth == 0 && use_color)
155 m_buffer << llvm::sys::Process::ResetColor();
156
157 if (start_pos == string::npos)
158 break;
159
160 auto angle = msg.at(start_pos);
161 m_template_depth += (angle == '<' || angle == '(') ? 1 : -1;
162 m_buffer << angle;
163 prev_pos = start_pos + 1;
164 }
165}

172void stacktrace_ostream::format_phase(string_view msg) {
173 static array skip_lines = {"yume::CRTPWalker<"sv, "__libc_start_"sv};
174 static string source_dir = YUME_SRC_DIR;
175
176 switch (m_current_phase) {
177 case Address: m_buffer << msg; break;
178 case Index:
179 set_color(GREEN);
180 m_buffer << msg;
181 break;
182 case Offset:
183 set_color(RED);
184 m_buffer << msg;
185 break;
186 case Function:
187 set_color(CYAN);
188 if (std::ranges::any_of(skip_lines, [&](string_view s) { return msg.find(s) != string::npos; }))
189 m_skip = true;
190 else
191 simplify(msg);
192 break;
193 case Source:
194 set_color(YELLOW);
195 if (auto normal_path = std::filesystem::path(msg).lexically_normal().native();
196 normal_path.starts_with(source_dir) && use_color) {
197 m_buffer << llvm::sys::Process::OutputBold(false);
198 m_buffer << static_cast<string_view>(normal_path).substr(source_dir.size());
199 } else {
200 m_buffer << normal_path;
201 }
202 };
203
204 if (use_color)
205 m_buffer << llvm::sys::Process::ResetColor();
206}

208void stacktrace_ostream::write_impl(const char* ptr, size_t size) {
209 auto msg = string_view(ptr, size);
210 m_total_size += size;
211
212 if (size == 1 && llvm::isSpace(msg.at(0))) {
213 m_buffer << msg.at(0);
214 return;
215 }
216
217 if (m_unknown) {
218 m_unknown = msg.at(0) != ')';
219 } else if (msg.at(0) == '(') {
220 m_skip = m_current_phase == Function;
221 m_unknown = true;
222 m_current_phase = Offset;
223 } else if (msg.at(0) == '/' && m_current_phase != Source) {
224 m_current_phase = Source;
225 }
226
227 format_phase(msg);
228
229 int counter = static_cast<int>(m_current_phase);
230 if (m_current_phase != Offset)
231 counter = ++counter % 4;
232
233 if (counter == 0)
234 clean_buffer();
235
236 m_current_phase = static_cast<Phase>(counter);
237 if (m_current_phase == Offset && !m_unknown)
238 m_current_phase = Index;
239}
