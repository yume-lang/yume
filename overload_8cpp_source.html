79static auto literal_cast(ast::AST& arg, ty::Type target_type) -> ty::Compat {
80 if (arg.val_ty() == target_type)
81 return {.valid = true}; // Already the correct type
82
83 if (isa<ast::NumberExpr>(arg) && target_type.base_isa<ty::Int>()) {
84 auto& num_arg = cast<ast::NumberExpr>(arg);
85 const auto* int_type = target_type.base_cast<ty::Int>();
86
87 if (int_type->size() == 1)
88 return {}; // Can't implicitly cast to Bool
89
90 auto in_range = int_type->in_range(num_arg.val);
91
92 if (in_range)
93 return {.valid = true, .conv = {.dereference = false, .kind = ty::Conv::Int}};
94 }
95
96 return {};
97}

99auto parameter_count_matches(const vector<ast::AST*>& args, const Fn& fn) -> bool {

110auto OverloadSet::is_valid_overload(Overload& overload) -> bool {
111 const auto& fn = *overload.fn;
112 const auto parent = fn.self_ty;
113
114 // Check if the call has a receiver matching the type of the struct this method is in.
115 // If the call has a receiver, it will always fail to match against against a top level function.
116 // Note that a receiver is always a type, such as `Foo.method`. Calls with an "object" as a receiver look similar,
117 // but `foo.method` is always rewritten to `method(foo)` and thus uses the "argument dependent lookup" rules below.
118 if (overload_receiver(call) != parent) {
119 if (!parent.has_value())
120 return false;
121
122 // If there is no matching receiver, check if any arguments are of the type of the struct.
123 // This perform "argument dependent lookup" and is required for "member functions"
124 if (std::ranges::none_of(
125 args, [parent](ast::AST* ast) { return ast->ensure_ty().without_mut().without_opaque() == *parent; })) {
126 notes.emit(overload.location()) << "Overload not considered due to ADL";
127 for (const auto* ast : args) {
128 notes.emit(overload.location()) << " Because `" << ast->ensure_ty().without_mut().without_opaque().name()
129 << "' is not `" << parent->name() << "'";
130 }
131 return false;
132 }
133 }
134
135 // The overload is only viable if the amount of arguments matches the amount of parameters.
136 if (!parameter_count_matches(args, fn)) {
137#ifdef YUME_SPEW_OVERLOAD_SELECTION
138 errs() << "!!! Unsuitable (parameter count): ";
139 overload.dump(errs());
140 errs() << "\n";
141#endif
142 return false;
143 }
144
145 overload.compatibilities.reserve(args.size());
146
147 // Determine the type compatibility of each argument individually. The performed conversions are also recorded for
148 // each step.
149 // As `llvm::zip` only iterates up to the size of the shorter argument, we don't try to determine type
150 // compatibility of the "variadic" part of varargs functions. Currently, varargs methods can only be primitives and
151 // carry no type information for their variadic part. This will change in the future.
152 for (const auto& [param_type_r, arg] : llvm::zip_first(fn.arg_types(), args)) {
153 auto arg_type = arg->val_ty();
154 optional<ty::Type> param_type = param_type_r;
155
156 if (param_type->is_generic()) {
157 auto sub = arg_type->determine_generic_subs(*param_type, overload.subs);
158
159 // No valid substitution found
160 if (!sub)
161 return false;
162
163 param_type = param_type->apply_generic_substitution(*sub);
164
165 if (!param_type)
166 return false;
167 }
168
169 // Attempt to do a literal cast
170 auto compat = literal_cast(*arg, *param_type);
171 // Couldn't perform a literal cast, try regular casts
172 if (!compat.valid)
173 compat = arg_type->compatibility(*param_type);
174
175 // Couldn't perform any kind of valid cast: one invalid conversion disqualifies the function entirely
176 if (!compat.valid) {
177 notes.emit(overload.location()) << "Overload not valid";
178 notes.emit(overload.location()) << " Because `" << arg_type->name() << "' is not convertible to `"
179 << param_type->name() << "'";
180 return false;
181 }
182
183 // Save the steps needed to perform the conversion
184 overload.compatibilities.push_back(compat);
185 }
186
187 // Add dummy conversions for each argument which maps to a variadic
188 while (overload.compatibilities.size() < args.size())
189 overload.compatibilities.emplace_back();
190
191 // Must be valid!
192 return true;
193}

202static auto cmp(bool a, bool b) -> std::strong_ordering { return static_cast<int>(a) <=> static_cast<int>(b); }

204static auto compare_implicit_conversions(ty::Conv a, ty::Conv b) -> std::weak_ordering {
205 const auto& equal = std::strong_ordering::equal;
206
207 // No conversion is better than some conversion
208 if (auto c = cmp(a.kind == ty::Conv::None, b.kind == ty::Conv::None); c != equal)
209 return c;
210
211 // No dereference is better than performing a dereference
212 if (auto c = cmp(!a.dereference, !b.dereference); c != equal)
213 return c;
214
215 // Cannot distinguish!
216 return equal;
217}

219auto Overload::better_candidate_than(Overload other) const -> bool {
220 // Viable candidates are always better than non-viable ones
221 if (!other.viable)
222 return viable;
223 if (!viable)
224 return false;
225
226 // For each argument, determine which candidate has a "better" conversion.
227 for (const auto& [self_compat, other_compat] : llvm::zip_first(compatibilities, other.compatibilities)) {
228 auto comparison = compare_implicit_conversions(self_compat.conv, other_compat.conv);
229
230 if (is_gt(comparison))
231 return true;
232 if (is_lt(comparison))
233 return false;
234
235 // Cannot distinguish between these, try the next arguments
236 }
237
238 // If we got to here, it means all arguments were identical. Neither overload is better than the other
239 return false;
240}

287 ss << "\nCouldn't pick between the following overloads:\n";
