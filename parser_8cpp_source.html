150auto Parser::parse_type(bool implicit_self) -> unique_ptr<Type> {
151 auto entry = tokens.begin();
152 auto base = [&]() -> unique_ptr<Type> {
153 if (implicit_self || try_consume(KWD_SELF_TYPE))
154 return ast_ptr<SelfType>(entry);
155
156 const string name = consume_word();
157 if (!is_uword(name))
158 throw std::runtime_error("Expected capitalized payload for simple type");
159
160 return ast_ptr<SimpleType>(entry, name);
161 }();
162 while (true) {
163 if (try_consume(KWD_PTR)) {
164 base = ast_ptr<QualType>(entry, move(base), Qualifier::Ptr);
165 } else if (try_consume(KWD_MUT)) {
166 base = ast_ptr<QualType>(entry, move(base), Qualifier::Mut);
167 } else if (try_peek(0, SYM_LBRACKET) && try_peek(1, SYM_RBRACKET)) {
168 // Don't consume the `[` unless the `]` is directly after; it might be a slice literal.
169 consume(SYM_LBRACKET);
170 consume(SYM_RBRACKET);
171 base = ast_ptr<QualType>(entry, move(base), Qualifier::Slice);
172 } else if (try_consume(SYM_LBRACE)) {
173 auto type_args = vector<AnyType>{};
174 consume_with_commas_until(SYM_RBRACE, [&] { type_args.emplace_back(parse_type()); });
175
176 base = ast_ptr<TemplatedType>(entry, move(base), move(type_args));
177 } else {
178 break;
179 }
180 }
181
182 return base;
183}

185auto Parser::try_parse_type() -> optional<unique_ptr<Type>> {
186 auto entry = tokens.begin();
187 if (tokens->type != Word || !tokens->payload.has_value())
188 return {};
189
190 const string name = consume_word();
191 if (make_atom(name) != std::get<Atom>(KWD_SELF_TYPE) && !is_uword(name))
192 return {};
193
194 unique_ptr<Type> base{};
195 if (make_atom(name) == std::get<Atom>(KWD_SELF_TYPE))
196 base = ast_ptr<SelfType>(entry);
197 else
198 base = ast_ptr<SimpleType>(entry, name);
199
200 while (true) {
201 if (try_consume(KWD_PTR)) {
202 base = ast_ptr<QualType>(entry, move(base), Qualifier::Ptr);
203 } else if (try_consume(KWD_MUT)) {
204 base = ast_ptr<QualType>(entry, move(base), Qualifier::Mut);
205 } else if (try_peek(0, SYM_LBRACKET) && try_peek(1, SYM_RBRACKET)) {
206 // Don't consume the `[` unless the `]` is directly after; it might be a slice literal.
207 consume(SYM_LBRACKET);
208 consume(SYM_RBRACKET);
209 base = ast_ptr<QualType>(entry, move(base), Qualifier::Slice);
210 } else if (try_consume(SYM_LBRACE)) {
211 auto type_args = vector<AnyType>{};
212 consume_with_commas_until(SYM_RBRACE, [&] { type_args.emplace_back(parse_type()); });
213
214 base = ast_ptr<TemplatedType>(entry, move(base), std::move(type_args));
215 } else {
216 break;
217 }
218 }
219
220 return base;

236auto Parser::parse_fn_name() -> string {
237 string name{};
238 if (tokens->type == Word) {
239 name = consume_word();
240 } else if (tokens->type == Symbol) {
241 // Try to parse an operator name, as in `def +()`
242 bool found_op = false;
243 for (const auto& op_row : operators()) {
244 for (const auto& op : op_row) {
245 if (try_consume(op)) {
246 found_op = true;
247 name = std::get<Atom>(op);
248 break;
249 }
250 }
251 if (found_op)
252 break;
253 }
254
255 // If an operator wasn't found, try parse the operator []
256 if (try_consume(SYM_LBRACKET)) {
257 consume(SYM_RBRACKET);
258 name = "[]";
259 } else if (try_consume(SYM_BANG)) {
260 name = "!"; // ! is unary, but the above operator check only checked binary ones
261 }
262 }
263
264 // Check if an equal sign follows, for fused assignement operators such as `+=` or `[]=`
265 if (try_consume(SYM_EQ))
266 name += "=";
267
268 return name;
269}

429auto Parser::parse_if_stmt() -> unique_ptr<IfStmt> {
430 auto entry = tokens.begin();
431 auto clause_begin = entry;
432 consume(KWD_IF);
433 auto cond = parse_expr();
434 if (!try_consume(KWD_THEN))
435 require_separator();
436
437 auto current_entry = tokens.begin();
438 auto else_entry = tokens.begin();
439 auto clauses = vector<IfClause>{};
440 auto current_body = vector<AnyStmt>{};
441 auto else_body = vector<AnyStmt>{};
442 bool in_else = false;
443
444 while (true) {
445 auto current_clause_begin = tokens.begin();
446 if (try_consume(KWD_END))
447 break;
448 if (try_consume(KWD_ELSE)) {
449 // An `else` followed by an `if` begins a new clause of the same if statement.
450 if (!in_else && try_consume(KWD_IF)) {
451 clauses.emplace_back(ts(clause_begin), move(cond), make_ast<Compound>(current_entry, move(current_body)));
452 current_body = vector<AnyStmt>{};
453 cond = parse_expr();
454 current_entry = tokens.begin();
455 clause_begin = current_clause_begin;
456 } else {
457 in_else = true;
458 else_entry = tokens.begin();
459 }
460 if (!try_consume(KWD_THEN))
461 require_separator();
462 }
463 auto st = parse_stmt();
464 if (in_else) {
465 else_body.emplace_back(move(st));
466 } else {
467 current_body.emplace_back(move(st));
468 }
469 }
470
471 if (else_body.empty())
472 else_entry = tokens.begin();
473
474 clauses.emplace_back(ts(clause_begin, else_entry - 1), move(cond),
475 Compound(ts(current_entry, else_entry - 1), move(current_body)));
476
477 auto else_clause = optional<Compound>{};
478 if (!else_body.empty())
479 else_clause.emplace(ts(else_entry), move(else_body));
480
481 return ast_ptr<IfStmt>(entry, move(clauses), move(else_clause));
482}

508auto Parser::parse_primary() -> unique_ptr<Expr> {
509 const auto guard = make_guard("Parsing primary expression");
510
511 auto entry = tokens.begin();
512 if (try_consume(SYM_LPAREN)) {
513 auto val = parse_expr();
514 consume(SYM_RPAREN);
515 return val;
516 }
517
518 if (tokens->type == Number)
519 return parse_number_expr();
520 if (tokens->type == Token::Type::Literal)
521 return parse_string_expr();
522 if (tokens->type == Token::Type::Char)
523 return parse_char_expr();
524 if (try_consume(KWD_TRUE))
525 return ast_ptr<BoolExpr>(entry, true);
526 if (try_consume(KWD_FALSE))
527 return ast_ptr<BoolExpr>(entry, false);
528 if (try_consume(SYM_COLON_COLON))
529 return ast_ptr<FieldAccessExpr>(entry, std::nullopt, consume_word());
530
531 if (tokens->type == Word) {
532 if (try_peek_uword(0)) {
533 auto type = parse_type();
534 if (try_consume(SYM_LPAREN)) {
535 auto call_args = vector<AnyExpr>{};
536 consume_with_commas_until(SYM_RPAREN, [&] { call_args.emplace_back(parse_expr()); });
537 return ast_ptr<CtorExpr>(entry, move(type), move(call_args));
538 }
539 if (try_consume(SYM_COLON)) {
540 if (try_consume(SYM_LBRACKET)) {
541 auto slice_members = vector<AnyExpr>{};
542 consume_with_commas_until(SYM_RBRACKET, [&] { slice_members.emplace_back(parse_expr()); });
543 return ast_ptr<SliceExpr>(entry, move(type), move(slice_members));
544 }
545 }
546
547 throw std::runtime_error("Couldn't make an expression from here with a type");
548 }
549 auto name = consume_word();
550 if (try_consume(SYM_LPAREN)) {
551 auto call_args = vector<AnyExpr>{};
552 consume_with_commas_until(SYM_RPAREN, [&] { call_args.emplace_back(parse_expr()); });
553 return ast_ptr<CallExpr>(entry, name, move(call_args));
554 }
555 return ast_ptr<VarExpr>({entry, 1}, name);
556 }
557 throw std::runtime_error("Couldn't make an expression from here");
558}

560auto Parser::parse_receiver(unique_ptr<Expr> receiver, VectorTokenIterator receiver_entry) -> unique_ptr<Expr> {
561 auto entry = tokens.begin();
562 if (try_consume(SYM_DOT)) {
563 auto name = consume_word();
564 auto call_args = vector<AnyExpr>{};
565 call_args.emplace_back(move(receiver));
566 if (try_consume(SYM_LPAREN)) { // A call with a dot `a.b(...)`
567 consume_with_commas_until(SYM_RPAREN, [&] { call_args.emplace_back(parse_expr()); });
568 auto call = ast_ptr<CallExpr>(entry + 1, name, move(call_args));
569 return parse_receiver(move(call), receiver_entry);
570 }
571 if (try_consume(SYM_EQ)) { // A setter `a.b = ...`
572 auto value = parse_expr();
573 call_args.emplace_back(move(value));
574 auto call = ast_ptr<CallExpr>(entry + 1, name + '=', move(call_args));
575 return parse_receiver(move(call), receiver_entry);
576 }
577 auto noarg_call = ast_ptr<CallExpr>(receiver_entry, name, move(call_args));
578 return parse_receiver(move(noarg_call), receiver_entry);
579 }
580 if (try_consume(SYM_EQ)) {
581 auto value = parse_expr();
582 auto assign = ast_ptr<AssignExpr>(receiver_entry, move(receiver), move(value));
583 return parse_receiver(move(assign), receiver_entry);
584 }
585 if (try_consume(SYM_LBRACKET)) {
586 auto args = vector<AnyExpr>{};
587 args.emplace_back(move(receiver));
588 args.emplace_back(parse_expr());
589 consume(SYM_RBRACKET);
590 if (try_consume(SYM_EQ)) {
591 auto value = parse_expr();
592 args.emplace_back(move(value));
593 auto call = ast_ptr<CallExpr>(entry, "[]=", move(args));
594 return parse_receiver(move(call), receiver_entry);
595 }
596 auto call = ast_ptr<CallExpr>(entry, "[]", move(args));
597 return parse_receiver(move(call), receiver_entry);
598 }
599 if (try_consume(SYM_COLON_COLON)) {
600 auto field = consume_word();
601 auto access = ast_ptr<FieldAccessExpr>(receiver_entry, move(receiver), field);
602 return parse_receiver(move(access), receiver_entry);
603 }
604 return receiver;
605}
