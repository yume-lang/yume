24/// An iterator-like holding `Token`s, used when parsing.
25/**
26 * Every parse method usually takes this as the first parameter. This is its own struct as it actually holds two
27 * iterators, where one is the end. This is to provide safe indexing (avoiding going past the end) without having to
28 * pass the end iterator around as a separate parameter.
29 */
30class TokenIterator {
31 VectorTokenIterator m_iterator;
32 VectorTokenIterator m_end;
33
34public:
35 constexpr TokenIterator(const VectorTokenIterator& iterator, const VectorTokenIterator& end)
36 : m_iterator{iterator}, m_end{end} {}
37
38 /// Check if the iterator is at the end and no more `Token`s could possibly be read.
39 [[nodiscard]] constexpr auto at_end() const noexcept -> bool { return m_iterator == m_end; }
40 [[nodiscard]] auto constexpr operator->() const -> Token* {
41 if (at_end())
42 throw std::runtime_error("Can't dereference at end");
43 return m_iterator.operator->();
44 }
45 [[nodiscard]] constexpr auto operator*() const -> Token {
46 if (at_end())
47 throw std::runtime_error("Can't dereference at end");
48 return m_iterator.operator*();
49 }
50 [[nodiscard]] constexpr auto operator+(int i) const noexcept -> TokenIterator {
51 return TokenIterator{m_iterator + i, m_end};
52 }
53 constexpr auto operator++() -> TokenIterator& {
54 if (at_end())
55 throw std::runtime_error("Can't increment past end");
56
57 ++m_iterator;
58 return *this;
59 }
60 auto operator++(int) -> TokenIterator {
61 if (at_end())
62 throw std::runtime_error("Can't increment past end");
63
64 return TokenIterator{m_iterator++, m_end};
65 }
66 /// Returns the underlying iterator. This shouldn't really be needed but I'm too lazy to properly model an iterator.
67 [[nodiscard]] auto begin() const -> VectorTokenIterator { return m_iterator; }
68};

263 template <size_t N = 0> auto parse_operator() -> unique_ptr<Expr> {
264 auto entry = tokens.begin();
265 const auto ops = operators();
266 if constexpr (N == ops.size()) {
267 return parse_unary();
268 } else {
269 auto left = parse_operator<N + 1>();
270 while (true) {
271 auto found_operator = false;
272 for (const auto& op : ops[N]) {
273 if (try_consume(op)) {
274 auto right = parse_operator<N + 1>();
275 auto args = vector<AnyExpr>{};
276 args.emplace_back(move(left));
277 args.emplace_back(move(right));
278 left = ast_ptr<CallExpr>(entry, string(std::get<Atom>(op)), move(args));
279 found_operator = true;
280 break;
281 }
282 }
283 if (!found_operator)
284 break;
285 }
286 return left;
287 }
