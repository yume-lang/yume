145/// Dereference an `optional` holding a pointer-like value, if it is holding a value
146/**
147 * Because C++20 doesn't have monadic operations for optionals yet, which should be a basic, fundamental feature of an
148 * "Optional" type
149 */
150template <typename T>
151requires pointer_like<T>
152auto inline constexpr try_dereference(const std::optional<T>& opt) {
153 using U = std::reference_wrapper<std::remove_reference_t<decltype(*opt.value())>>;
154 if (opt.has_value())
155 return std::optional<U>(*opt.value());
156 return std::optional<U>{};
157}
