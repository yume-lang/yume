45/// A criterion for classifying a stream of characters as a specific type of token \link Token::Type
46/**
47 * TODO: requires doc update regarding TokenState (and TokenState itself needs docs)
48 *
49 * The criterion is a function taking three arguments:
50 * an `int` representing the current character being evaluated, a second `int` representing the
51 * index of this character relative to the current character sequence (that is, since the end of the previous token),
52 * and a `stringstream`, where matching characters should be written to. The function should return a `bool`,
53 * whether or not this character is viable as this type of token. As soon as `false` is returned, this token ends and
54 * the process is repeated.
55 *
56 * Note that while usually the same character (the first `int` argument) should be appended, tokens such as string
57 * literals may include escapes and thus output the corresponding unescaped value.
58 */
59struct Characteristic {
60 char_fn fn;
61 Token::Type type;
62
63 /// Criterion doesn't take a stringstream: if `true` is returned the same character is appended. This is usually
64 /// preferred.
65 Characteristic(Token::Type type, const std::function<bool(int, int)>& fn)
66 : fn([fn](TokenState& state) {
67 bool b = fn(state.c, static_cast<int>(state.index));
68 if (b)
69 state.stream.put(state.c);
70 return b;
71 }),
72 type(type) {}
73 Characteristic(Token::Type type, char_fn fn) : fn(move(fn)), type(type) {}
74};

92 /// Strings are delimited by double quotes `"` and may contain escapes.
93 constexpr static const auto is_str = [end = false, escape = false](TokenState& state) mutable {
94 if (end)
95 return false;
96
97 state.valid = false;
98 if (state.index == 0)

104 end = true;
105 state.valid = true;

117 /// Chars begin with a question mark `?` and may contain escapes.
118 constexpr static const auto is_char = [escape = false](TokenState& state) mutable {

145 /// Generate a criterion matching a single character from any within the string `chars`.
146 constexpr static const auto is_any_of = [](string chars) {

150 /// Generate a criterion matching the string exactly.
151 constexpr static const auto is_exactly = [](string str) {
152 return [s = move(str)](TokenState& state) {
153 if (state.index == s.size())
154 return false;
155 state.valid = state.index == s.size() - 1;
156
157 bool matches = s[state.index] == state.c;
158 if (matches) {
159 state.stream.put(state.c);
160 }
161
162 return matches;
163 };

171 void tokenize() {
172
173 while (!m_in.eof()) {
174 if (!select_characteristic({
175 {Token::Type::Separator, is_exactly("\n")},
176 {Token::Type::Skip, is_c(isspace)},
177 {Token::Type::Skip, is_comment},
178 {Token::Type::Number, is_c(isdigit)},
179 {Token::Type::Literal, is_str},
180 {Token::Type::Char, is_char},
181 {Token::Type::Word, is_word},
182 {Token::Type::Symbol, is_exactly("==")},
183 {Token::Type::Symbol, is_exactly("!=")},
184 {Token::Type::Symbol, is_exactly("//")},
185 {Token::Type::Symbol, is_any_of(R"(()[]{}<>=:#%-+.,!/*\)")},
186 })) {
187 string message = "Tokenizer didn't recognize ";
188 message += m_last;
189 throw std::runtime_error(message);
190 }
191 m_count++;
192 }
193 }

211 /// Find the first criterion for which the current character is viable as the first character, then consume tokens
212 /// until the criterion becomes false. The result is appended to the current list of tokens `m_tokens`.
213 ///
214 /// \returns `false` if no criterion matched.
215 /// \sa is_characteristic, consume_characteristic
216 auto select_characteristic(std::initializer_list<Characteristic> list) -> bool {
217 int begin_line = m_line;
218 int begin_col = m_col;
219 char begin_last = m_last;
220 auto begin_position = m_in.tellg();
221
222 return any_of(list, [&](const auto& c) {
223 m_line = begin_line;
224 m_col = begin_col;
225 m_last = begin_last;
226 m_in.seekg(begin_position);
227 m_in.clear();
228
229 auto stream = stringstream{};
230 auto state = TokenState{true, m_last, 0, stream};
231 if (c.fn(state)) {
232 auto [atom, end_line, end_col] = consume_characteristic(c.fn, state);
233 if (state.valid) {
234 m_tokens.emplace_back(c.type, atom, m_count, Loc{begin_line, begin_col, end_line, end_col, m_source_file});
235 return true;
236 }
237 }
238 return false;
239 });
240 }

242 /// Consume characters until the criterion becomes false. Note that the first character is assumed to already be
243 /// matched by `is_characteristic`
244 /// \returns `Atom` containing the payload of the matched token, and the line and col number it stopped on.
245 auto consume_characteristic(const char_fn& fun, TokenState& state) -> std::tuple<Atom, int, int> {
246 state.index++;
247 int end_line = m_line;
248 int end_col = m_col;
249 next();
250 state.c = m_last;
251 while (!m_in.eof() && fun(state)) {
252 state.index++;
253 end_line = m_line;
254 end_col = m_col;
255 next();
256 state.c = m_last;
257 }
258
259 return {make_atom(state.stream.str()), end_line, end_col};
260 }
