37/// A criterion for classifying a stream of characters as a specific type of token \link Token::Type
38/**
39 * The criterion is a function taking three arguments:
40 * an `int` representing the current character being evaluated, a second `int` representing the
41 * index of this character relative to the current character sequence (that is, since the end of the previous token),
42 * and a `std::stringstream`, where matching characters should be written to. The function should return a `bool`,
43 * whether or not this character is viable as this type of token. As soon as `false` is returned, this token ends and
44 * the process is repeated.
45 *
46 * Note that while usually the same character (the first `int` argument) should be appended, tokens such as string
47 * literals may include escapes and thus output the corresponding unescaped value.
48 */
49struct Characteristic {
50 char_fn m_fn;
51 Token::Type m_type;
52
53 /// Criterion doesn't take a stringstream: if `true` is returned the same character is appended. This is usually
54 /// preferred.
55 Characteristic(Token::Type type, const std::function<bool(int, int)>& fn)
56 : m_fn([fn](int c, int i, std::stringstream& stream) {
57 bool b = fn(c, i);
58 if (b)
59 stream.put((char)c);
60 return b;
61 }),
62 m_type(type) {}
63 Characteristic(Token::Type type, char_fn fn) : m_fn(std::move(fn)), m_type(type) {}
64};

76 /// Words consist of alphanumeric characters, or underscores, but *must* begin with a letter.
77 constexpr static const auto is_word = [](int c, int i) {

81 /// Strings are delimited by double quotes `"` and may contain escapes.
82 constexpr static const auto is_str = [end = false, escape = false](char c, int i, std::stringstream& stream) mutable {
83 if (i == 0)
84 return c == '"';
85 if (end)
86 return false;
87
88 if (c == '\\' && !escape) {
89 escape = true;
90 } else if (c == '"' && !escape && !end) {
91 end = true;

93 if (escape && c == 'n')
94 stream.put('\n');
95 else
96 stream.put((char)c);
97
98 escape = false;
99 }
100 return true;
101 };

103 /// Chars begin with a question mark `?` and may contain escapes.
104 constexpr static const auto is_char = [escape = false](char c, int i, std::stringstream& stream) mutable {
105 if (i == 0)
106 return c == '?';
107 if (i == 1) {
108 if (c == '\\')
109 escape = true;
110 else
111 stream.put((char)c);
112 return true;
113 }
114 if (i == 2 && escape) {

117 else
118 stream.put((char)c);

120 }
121 return false;

124 /// Comments begin with an octothorpe `#` and last until the end of the line.
125 constexpr static const auto is_comment = [](char c, int i) { return (i == 0 && c == '#') || (i > 0 && c != '\n'); };

127 /// Generate a criterion matching a single character from any within the string `chars`.
128 constexpr static const auto is_any_of = [](string chars) {
129 return [checks = move(chars)](char c, int i) { return i == 0 && checks.find(c) != string::npos; };

143 void tokenize() {
144
145 while (!m_in.eof()) {
146 if (!select_characteristic({
147 {Token::Type::Separator, is_exactly("\n")},
148 {Token::Type::Skip, is_c(isspace)},
149 {Token::Type::Skip, is_comment},
150 {Token::Type::Number, is_c(isdigit)},
151 {Token::Type::Literal, is_str},
152 {Token::Type::Char, is_char},
153 {Token::Type::Word, is_word},
154 {Token::Type::Symbol, is_exactly("==")},
155 {Token::Type::Symbol, is_exactly("!=")},
156 {Token::Type::Symbol, is_exactly("//")},
157 {Token::Type::Symbol, is_any_of(R"(()[]<>=:#"%-+.,!?/*\)")},
158 })) {
159 string message = "Tokenizer didn't recognize ";
160 message += m_last;
161 throw std::runtime_error(message);
162 }
163 m_count++;
164 }

181 /// Check if the current character is viable as the first character of the given criterion.
182 [[nodiscard]] auto is_characteristic(const char_fn& fun, std::stringstream& stream) const -> bool {
183 return fun(m_last, 0, stream);

186 /// Find the first criterion for which the current character is viable as the first character, then consume tokens
187 /// until the criterion becomes false. The result is appended to the current list of tokens `m_tokens`.
188 ///
189 /// \returns `false` if no criterion matched.
190 /// \sa is_characteristic, consume_characteristic
191 auto select_characteristic(std::initializer_list<Characteristic> list) -> bool {
192 int begin_line = m_line;
193 int begin_col = m_col;
194 return any_of(list, [&](const auto& c) {
195 auto stream = std::stringstream{};
196 if (is_characteristic(c.m_fn, stream)) {
197 auto [atom, end_line, end_col] = consume_characteristic(c.m_fn, stream);
198 m_tokens.emplace_back(c.m_type, atom, m_count, Loc{begin_line, begin_col, end_line, end_col, m_source_file});
199 return true;
200 }
201 return false;
202 });
203 }

205 /// Consume characters until the criterion becomes false. Note that the first character is assumed to already be
206 /// matched by `is_characteristic`
207 /// \returns `Atom` containing the payload of the matched token, and the line and col number it stopped on.
208 auto consume_characteristic(const char_fn& fun, std::stringstream& out) -> std::tuple<Atom, int, int> {
209 int i = 1;
210 int end_line = m_line;
211 int end_col = m_col;
212 next();
213 while (!m_in.eof() && fun(m_last, i, out)) {
214 i++;
215 end_line = m_line;
216 end_col = m_col;
217 next();
218 }
219
220 return {make_atom(out.str()), end_line, end_col};
221 }
