15auto Type::known_qual(Qualifier qual) const -> const Type& {
16 int qual_idx = static_cast<int>(qual);
17 const auto& existing = m_known_qual.at(qual_idx);
18 if (existing != nullptr)
19 return *existing;
20
21 const auto* base = qual_base();
22 // Qualifiers like Mut can't repeat, so attempting to get mut of a mut type should return itself
23 if (qual == Qualifier::Mut) {
24 bool existing_mut = false;
25 const auto* existing_base = this;
26 if (base != nullptr) {
27 if (base->m_known_qual.at(qual_idx).get() == this)
28 return *this;
29 const auto* existing_qual = cast<Qual>(this);
30 existing_mut = existing_qual->m_mut;
31 existing_base = &existing_qual->m_base;
32 }
33 auto new_qual_type =
34 std::make_unique<Qual>(name() + qual_suffix(qual), *existing_base, (existing_mut || qual == Qualifier::Mut));
35 m_known_qual.at(qual_idx) = move(new_qual_type);
36 } else {
37 auto new_qual_type = std::make_unique<Ptr>(name() + qual_suffix(qual), *this, qual);
38 m_known_qual.at(qual_idx) = move(new_qual_type);
39 }
40 return *m_known_qual.at(qual_idx);
41}

43auto Type::determine_generic_substitution(const Type& generic, Sub sub) const -> Sub {
44 yume_assert(generic.is_generic(), "Cannot substitute generics in a non-generic type");
45
46 // `Foo ptr` -> `T ptr`, with `T = Foo`.
47 if (auto this_ptr_base = ptr_base(), gen_ptr_base = generic.ptr_base();
48 this_ptr_base != nullptr && gen_ptr_base != nullptr &&
49 cast<Ptr>(this)->qualifier() == cast<Ptr>(generic).qualifier()) {
50 return this_ptr_base->determine_generic_substitution(*gen_ptr_base, sub);
51 }
52 // `Foo mut` -> `T mut`, with `T = Foo`.
53 if (is_mut() && generic.is_mut())
54 return qual_base()->determine_generic_substitution(generic.without_qual(), sub);
55
56 // `Foo[] mut` -> `T[]`, with `T = Foo`.
57 if (is_mut() && !generic.is_mut())
58 return qual_base()->determine_generic_substitution(generic, sub);
59
60 // Substitution impossible! For example, `Foo` -> `T ptr`.
61 if (!isa<Generic>(generic))
62 return sub;
63
64 // Any other generic that didn't match above.
65 // `Foo ptr` -> `T`, with `T = Foo ptr`.
66 sub.target = &cast<Generic>(generic);
67 sub.replace = this;
68 return sub;
69}

71auto Type::compatibility(const Type& other, Compat compat) const -> Compat {
72 if (this == &other) {
73 compat.valid = true;
74 return compat;
75 }
76
77 // `Foo mut` -> `Foo`.
78 // Note that the base types are also compared, so `I32 mut` -> `I64`.
79 if (is_mut() && !other.is_mut()) {
80 compat.conv.dereference = true;
81 compat = qual_base()->compatibility(other.without_qual(), compat);
82 return compat;
83 }
84
85 // `I32` -> `I64`. `U8` -> `I16`. An implicit integer cast with no loss of information.
86 if (const auto this_int = dyn_cast<Int>(this), other_int = dyn_cast<Int>(&other);
87 (this_int != nullptr) && (other_int != nullptr)) {
88 if ((this_int->is_signed() == other_int->is_signed() && this_int->size() <= other_int->size()) ||
89 (!this_int->is_signed() && other_int->is_signed() && this_int->size() * 2 <= other_int->size())) {
90 compat.valid = true;
91 compat.conv.kind = Conv::Int;
92 return compat;
93 }
94 }
95 return compat;
96}

113auto Type::apply_generic_substitution(Sub sub) const -> const Type* {
114 yume_assert(is_generic(), "Can't perform generic substitution without a generic type");
115 yume_assert(sub.target != nullptr && sub.replace != nullptr,
