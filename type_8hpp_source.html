34/// Represents a type in the type system. All types inherit from this class.
35/**
36 * Similar to `AST` nodes, `Type`s cannot be copied or moved as all special member functions (aside from the destructor)
37 * are `delete`d. There will always be one instance of a `Type` object for a specific type. This means that comparing if
38 * two types are the same type is simply a pointer equality check. The single instance of most types is located in
39 * `TypeHolder`, except for "qualified" types, which are located in their base type.
40 *
41 * This means that a type such as `I32 ptr`, which is a `Ptr`, is stored in the `m_known_qual` array of the `I32` type,
42 * at the index `Qualifier::Ptr`.
43 */
44class Type {
45 mutable array<unique_ptr<Type>, static_cast<int>(Qualifier::Q_END)> m_known_qual{};
46 const Kind m_kind;
47 string m_name;
48
49public:
50 Type(const Type&) noexcept = delete;
51 Type(Type&&) noexcept = delete;
52 auto operator=(const Type&) noexcept -> Type& = delete;
53 auto operator=(Type&&) noexcept -> Type& = delete;
54 virtual ~Type() = default;
55 [[nodiscard]] auto kind() const -> Kind { return m_kind; };
56 [[nodiscard]] auto name() const -> string { return m_name; };
57
58 /// Get this type with a given qualifier applied.
59 /**
60 * This may construct a new `Qual` or `Ptr` type, or use a cached instance from `m_known_qual`. Note that since some
61 * qualifiers, such as `mut` don't stack. Getting the `mut`-qualified type of `T mut` returns itself.
62 */
63 [[nodiscard]] auto known_qual(Qualifier qual) const -> const Type&;
64 [[nodiscard]] auto known_ptr() const -> const Type& { return known_qual(Qualifier::Ptr); }
65 [[nodiscard]] auto known_mut() const -> const Type& { return known_qual(Qualifier::Mut); }
66 [[nodiscard]] auto known_slice() const -> const Type& { return known_qual(Qualifier::Slice); }
67
68 [[nodiscard]] auto determine_generic_substitution(const Type& generic, Sub sub = Sub()) const -> Sub;
69 [[nodiscard]] auto apply_generic_substitution(Sub sub) const -> const Type*;
70 [[nodiscard]] auto fully_apply_instantiation(const Instantiation& inst) const -> const Type*;
71 [[nodiscard]] auto compatibility(const Type& other, Compat compat = Compat()) const -> Compat;
72 /// The union of this and `other`. For example, the union of `T` and `T mut` is `T mut`.
73 /// \returns `nullptr` if an union cannot be created.
74 [[nodiscard]] auto coalesce(const Type& other) const -> const Type*;
75 /// Return the intersection of this and `other`. For example, the intersection of `T and `T mut` is `T`.
76 /// \returns `nullptr` is there's not intersection between the types.
77 [[nodiscard]] auto intersect(const Type& other) const -> const Type*;
78
79 [[nodiscard]] auto is_mut() const -> bool;
80 [[nodiscard]] auto is_generic() const -> bool;
81
82 /// If this type is a `Qual`, return the base of it (`T mut` -> `T`)
83 /// \returns `nullptr` if this type isn't `Qual`.
84 [[nodiscard]] auto qual_base() const -> const Type*;
85
86 /// If this type is a `Ptr`, return the base of it (`T ptr` -> `T`)
87 /// \returns `nullptr` if this type isn't `Ptr`.
88 [[nodiscard]] auto ptr_base() const -> const Type*;
89
90 /// If this type is a `Qual`, return the base of it, otherwise return itself.
91 [[nodiscard]] auto without_qual() const -> const Type&;
92 [[nodiscard]] auto without_qual_kind() const -> Kind;
93
94protected:
95 Type(Kind kind, string name) : m_kind(kind), m_name(move(name)) {}
96};
