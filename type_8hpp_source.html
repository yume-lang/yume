39/// Represents a type in the type system. All types inherit from this class.
40/**
41 * Similar to `AST` nodes, `Type`s cannot be copied or moved as all special member functions (aside from the destructor)
42 * are `delete`d. There will always be one instance of a `Type` object for a specific type. This means that comparing if
43 * two types are the same type is simply a pointer equality check. The single instance of most types is located in
44 * `TypeHolder`, except for "qualified" types, which are located in their base type.
45 *
46 * This means that a type such as `I32 ptr`, which is a `Ptr`, is stored in the `m_known_qual` array of the `I32` type,
47 * at the index `Qualifier::Ptr`.
48 */
49class Type {
50 mutable std::array<unique_ptr<Type>, static_cast<int>(Qualifier::Q_END)> m_known_qual{};
51 const Kind m_kind;
52 string m_name;
53
54public:
55 /// The compatibility between two types, for overload selection.
56 struct Compatiblity {
57 static constexpr const uint64_t INVALID = -1;
58
59 uint64_t rating{};
60 const Generic* substituted_generic{};
61 const Type* substituted_with{};
62
63 auto operator+(uint64_t other) const -> Compatiblity {
64 return {rating + other, substituted_generic, substituted_with};
65 }
66 };

68 Type(const Type&) noexcept = delete;
69 Type(Type&&) noexcept = delete;
70 auto operator=(const Type&) noexcept -> Type& = delete;
71 auto operator=(Type&&) noexcept -> Type& = delete;
72 virtual ~Type() = default;
73 [[nodiscard]] auto kind() const -> Kind { return m_kind; };
74 [[nodiscard]] auto name() const -> string { return m_name; };

76 /// Get this type with a given qualifier applied.
77 /**
78 * This may construct a new `Qual` or `Ptr` type, or use a cached instance from `m_known_qual`. Note that since some
79 * qualifiers, such as `mut` don't stack. Getting the `mut`-qualified type of `T mut` returns itself.
80 */
81 [[nodiscard]] auto known_qual(Qualifier qual) const -> const Type&;
82 [[nodiscard]] auto known_ptr() const -> const Type& { return known_qual(Qualifier::Ptr); }
83 [[nodiscard]] auto known_mut() const -> const Type& { return known_qual(Qualifier::Mut); }
84 [[nodiscard]] auto known_slice() const -> const Type& { return known_qual(Qualifier::Slice); }
85
86 [[nodiscard]] auto compatibility(const Type& other) const -> Compatiblity;

88 /// \returns `nullptr` if an union cannot be created.

97 /// \returns `nullptr` if this type isn't `Qual`.

109 Type(Kind kind, string name) : m_kind(kind), m_name(move(name)) {}
