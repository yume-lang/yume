27/// Represents a type in the type system.
28/// NOTE: that this isn't the class to use for type introspection, for that, `Type` should be used instead as it is
29/// fully qualified.
30///
31/// All kinds of types inherit from this class. Similar to `AST` nodes, `Type`s cannot be copied or moved as all special
32/// member functions (aside from the destructor) are `delete`d.
33///
34/// There will always be one instance of a `Type` object for a specific type. This means that comparing if two types are
35/// the same type is simply a pointer equality check.
36///
37/// The single instance of most types is located in `TypeHolder`, however "qualified" types are instead stored in
38/// instances of `Type`. A "qualified" type is, for example, `mut`. These qualifiers do not stack. Getting the "mut"
39/// reference to a mutable reference should be itself.
40///
41/// Also, pointer-like types are also not held in `TypeHolder`, and are instead stored in the `m_known_...` fields
42/// of their pointee type.
43class BaseType {
44 mutable unique_ptr<BaseType> m_known_ptr{};
45 mutable unique_ptr<BaseType> m_known_opaque_self{};
46 mutable unique_ptr<BaseType> m_known_meta{};
47 const Kind m_kind;
48 string m_name;
49
50 friend Type;
51
52public:
53 BaseType(const BaseType&) noexcept = delete;
54 BaseType(BaseType&&) noexcept = delete;
55 auto operator=(const BaseType&) noexcept -> BaseType& = delete;
56 auto operator=(BaseType&&) noexcept -> BaseType& = delete;
57 virtual ~BaseType() = default;
58 [[nodiscard]] auto kind() const -> Kind { return m_kind; };
59 [[nodiscard]] auto base_name() const -> string { return m_name; };
60 [[nodiscard]] virtual auto name() const -> string = 0;
61
62protected:
63 BaseType(Kind kind, string name) : m_kind(kind), m_name(move(name)) {}
64};
