23/// Represents a type in the type system.
24/// NOTE: that this isn't the class to use for type introspection, for that, `Type` should be used instead as it is
25/// fully qualified.
26///
27/// All kinds of types inherit from this class. Similar to `AST` nodes, `Type`s cannot be copied or moved as all special
28/// member functions (aside from the destructor) are `delete`d.
29///
30/// There will always be one instance of a `Type` object for a specific type. This means that comparing if two types are
31/// the same type is simply a pointer equality check.
32///
33/// The single instance of most types is located in `TypeHolder`, however "qualified" types are instead stored in
34/// instances of `Type`. A "qualified" type is, for example, `mut`. These qualifiers do not stack. Getting the "mut"
35/// reference to a mutable reference should be itself.
36///
37/// Also, pointer-like types are also not held in `TypeHolder`, and are instead stored in the `m_known_ptr_like` array
38/// of their pointee type.
39class BaseType {
40 mutable array<unique_ptr<BaseType>, static_cast<int>(PtrLikeQualifier::Q_END)> m_known_ptr_like{};
41 const Kind m_kind;
42 string m_name;
43
44 friend Type;
45
46public:
47 BaseType(const BaseType&) noexcept = delete;
48 BaseType(BaseType&&) noexcept = delete;
49 auto operator=(const BaseType&) noexcept -> BaseType& = delete;
50 auto operator=(BaseType&&) noexcept -> BaseType& = delete;
51 virtual ~BaseType() = default;
52 [[nodiscard]] auto kind() const -> Kind { return m_kind; };
53 [[nodiscard]] auto base_name() const -> string { return m_name; };
54 [[nodiscard]] virtual auto name() const -> string = 0;
55
56protected:
57 BaseType(Kind kind, string name) : m_kind(kind), m_name(move(name)) {}
58};
