35inline void try_implicit_conversion(ast::OptionalExpr& expr, optional<ty::Type> target_ty) {
36 if (!target_ty)

519auto TypeWalker::convert_type(const ast::Type& ast_type) -> ty::Type {
520 auto parent = current_decl.self_ty();
521 const auto* context = current_decl.subs();
522
523 if (const auto* simple_type = dyn_cast<ast::SimpleType>(&ast_type)) {
524 auto name = simple_type->name();
525 if (context != nullptr) {
526 auto generic = context->find(name);
527 if (generic != context->end())
528 return generic->second;
529 }
530 auto val = compiler.m_types.known.find(name);
531 if (val != compiler.m_types.known.end())
532 return val->second.get();
533 } else if (const auto* qual_type = dyn_cast<ast::QualType>(&ast_type)) {
534 auto qualifier = qual_type->qualifier();
535 if (qualifier == Qualifier::Slice)
536 return convert_slice_type(qual_type->base());
537 return convert_type(qual_type->base()).known_qual(qualifier);
538 } else if (isa<ast::SelfType>(ast_type)) {
539 if (parent)
540 return *parent;
541 }
542
543 throw std::runtime_error("Cannot convert AST type to actual type! "s + ast_type.kind_name() + " (" +
544 ast_type.describe() + ")");
545}

547void TypeWalker::resolve_queue() {
548 while (!decl_queue.empty()) {
549 auto next = decl_queue.front();
550 decl_queue.pop();
551
552 with_saved_scope([&] {
553 next.visit_decl(
554 [&](Fn* fn) {
555 in_depth = true;
556 compiler.declare(*fn);
557 },
558 [&](Struct* st) {
559 for (auto& i : st->body().body()) {
560 auto decl = compiler.decl_statement(*i, st->self_ty, st->member);
561
562 // "Inherit" substitutions
563 // TODO(rymiel): Shadowing type variables should be an error
564 for (auto& [k, v] : st->subs)
565 decl.subs()->try_emplace(k, v);
566 compiler.walk_types(decl);
567 }
568 },
569 [&](Ctor* /*ctor*/) { throw std::runtime_error("Unimplemented"); });
570 });
571 }
