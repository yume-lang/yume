84template <> void TypeWalker::expression(ast::FieldAccessExpr& expr) {
85 body_expression(expr.base());
86 const auto& type = *expr.base().val_ty();
87
88 const auto* struct_type = dyn_cast<ty::Struct>(&type.without_qual());
89
90 if (struct_type == nullptr)
91 throw std::runtime_error("Can't access field of expression with non-struct type");
92
93 auto target_name = expr.field();
94 const ty::Type* target_type{};
95 int j = 0;
96 for (const auto& field : struct_type->fields()) {
97 if (field.name() == target_name) {
98 target_type = field.type().val_ty();
99 break;
100 }
101 j++;
102 }
103
104 expr.offset(j);

118template <> void TypeWalker::expression(ast::CallExpr& expr) {
119 auto fns_by_name = vector<Fn*>();
120 auto overloads = vector<std::tuple<uint64_t, Fn*, Instantiation>>();
121 auto name = expr.name();
122
123 for (auto& fn : m_compiler.m_fns)
124 if (fn.name() == name)
125 fns_by_name.push_back(&fn);
126
127 if (fns_by_name.empty())
128 throw std::logic_error("No function overload named "s + name);
129
130 vector<const ty::Type*> arg_types{};
131 for (auto& i : expr.args()) {
132 body_expression(i);
133 arg_types.push_back(i.val_ty());
134 }
135
136#ifdef YUME_SPEW_OVERLOAD_SELECTION
137 errs() << "\n*** BEGIN OVERLOAD EVALUATION ***\n";
138 errs() << "Functions with matching names for call " << name << " with types ";
139 join_args(arg_types);
140 errs() << "\n";
141 for (const auto& i_s : fns_by_name) {
142 errs() << " " << i_s->name() << ":" << i_s->ast().location().begin_line << " with types ";
143 join_args(i_s->ast().args(), [](const auto& ast) { return ast.val_ty(); });
144 errs() << "\n";
145 }
146#endif
147
148 for (auto* fn : fns_by_name) {
149 uint64_t compat = 0;
150 Instantiation instantiation{};
151 const auto& fn_ast = fn->m_ast_decl;
152 auto fn_arg_size = fn_ast.args().size();
153
154 // The overload is only considered if it has a matching amount of parameters.
155 if (arg_types.size() == fn_arg_size || (expr.args().size() >= fn_arg_size && fn_ast.varargs())) {
156 // Determine the type compatibility of each argument individually.
157 unsigned i = 0;
158 for (const auto& arg_type : arg_types) {
159 // Don't try to determine type compatibility of the "var" part of varargs functions.
160 // Currently, varargs methods can only be primitives and carry no type information for their variadic part. This
161 // will change in the future.
162 if (i >= fn_arg_size)
163 break;
164
165 const auto* ast_arg = fn_ast.args()[i].val_ty();
166 auto [i_compat, gen, gen_sub] = arg_type->compatibility(*ast_arg);
167 // One incompatible parameter disqualifies the function entirely
168 if (i_compat == ty::Type::Compatiblity::INVALID) {
169 compat = i_compat;
170 break;
171 }
172
173 // The function compatibility determined a substitution for a type variable.
174 if (gen != nullptr && gen_sub != nullptr) {
175 auto existing = instantiation.m_sub.find(gen);
176 // The substitution must have an intersection with an already deduced value for the same type variable
177 if (existing != instantiation.m_sub.end()) {
178 const auto* intersection = gen_sub->intersect(*existing->second);
179 if (intersection == nullptr) {
180 // The types don't have a common intersection, they cannot coexist.
181 compat = ty::Type::Compatiblity::INVALID;
182 break;
183 }
184 instantiation.m_sub[gen] = intersection;
185 } else {
186 instantiation.m_sub.try_emplace(gen, gen_sub);
187 }
188 }
189 compat += i_compat;
190 i++;
191 }
192 if (compat != ty::Type::Compatiblity::INVALID)
193 overloads.emplace_back(compat, fn, instantiation);
194 }
195 }
196
197#ifdef YUME_SPEW_OVERLOAD_SELECTION
198 errs() << "Available overloads for call " << name << " with types ";
199 join_args(arg_types);
200 errs() << "\n";
201 for (const auto& [i_w, i_s, i_i] : overloads) {
202 errs() << " Weight " << i_w << " " << i_s->name() << ":" << i_s->ast().location().begin_line << " with types ";
203
204 join_args(i_s->ast().args(), [](const auto& ast) { return ast.val_ty(); });
205 errs() << " where";
206 for (const auto& [sub_k, sub_v] : i_i.m_sub) {
207 errs() << " " << sub_k->name() << " = " << sub_v->name();
208 }
209 errs() << "\n";
210 }
211 errs() << "\n*** END OVERLOAD EVALUATION ***\n\n";
212#endif
213
214 if (overloads.empty()) {
215 string str{};
216 raw_string_ostream ss{str};
217 ss << "No matching overload for " << name << " with argument types ";
218 join_args(arg_types, {}, ss);
219 throw std::logic_error(str);
220 }
221
222 auto [selected_weight, selected, instantiate] =
223 *std::max_element(overloads.begin(), overloads.end(),
224 [&](const auto& a, const auto& b) { return get<uint64_t>(a) < get<uint64_t>(b); });
225
226 // It is an instantiation of a function template
227 if (!instantiate.m_sub.empty()) {
228 // TODO: move most of this logic directly into Fn
229 // Try to find an already existing instantiation with the same substitutions
230 auto existing_instantiation = m_current_fn->m_instantiations.find(instantiate);
231 if (existing_instantiation == m_current_fn->m_instantiations.end()) {
232 // An existing one wasn't found. Duplicate the template's AST with the substituted types.
233 auto* decl_clone = selected->m_ast_decl.clone();
234 m_current_fn->m_member->direct_body().emplace_back(decl_clone);
235
236 std::map<string, const ty::Type*> subs{};
237 for (const auto& [k, v] : instantiate.m_sub)
238 subs.try_emplace(k->name(), v);
239
240 auto fn_ptr = std::make_unique<Fn>(*decl_clone, selected->m_parent, selected->m_member, move(subs));
241 auto new_emplace = m_current_fn->m_instantiations.emplace(instantiate, move(fn_ptr));
242 auto& new_fn = *new_emplace.first->second;
243
244 // TODO: Push this in a queue to not cause a huge stack trace when an instantiation causes another instantiation
245 // TODO: This really needs to go into a queue because an instantiation can loop back to requiring to instantiate
246 // itself while the first instantiation hasn't finished existing this scope yet, leading to god knows what
247 // outcome.
248
249 // Save everything pertaining to the old context
250 auto saved_scope = m_scope;
251 auto* saved_current = m_current_fn;
252
253 // Temporarily create a new context
254 m_in_depth = false;
255 m_scope.clear();
256 m_current_fn = &new_fn;
257
258 // Type-annotate the cloned declaration
259 body_statement(*decl_clone);
260 m_in_depth = true;
261
262 auto* llvm_fn = m_compiler.declare(new_fn);
263 new_fn.m_llvm_fn = llvm_fn;
264 selected = &new_fn;
265
266 // Restore again
267 m_scope = saved_scope;
268 m_current_fn = saved_current;
269 } else {
270 selected = existing_instantiation->second.get();
271 }
272 }
273
274 if (selected->m_ast_decl.ret().has_value())
275 expr.val_ty(selected->ast().ret()->get().val_ty());
276
277 expr.selected_overload(selected);
278}

354auto TypeWalker::convert_type(const ast::Type& ast_type) -> const ty::Type& {
355 const ty::Type* parent = m_current_fn == nullptr ? m_current_struct : m_current_fn->parent();
356 Fn* context = m_current_fn;
357
358 if (const auto* simple_type = dyn_cast<ast::SimpleType>(&ast_type)) {
359 auto name = simple_type->name();
360 if (context != nullptr) {
361 auto generic = context->m_subs.find(name);
362 if (generic != context->m_subs.end())
363 return *generic->second;
364 }
365 auto val = m_compiler.m_types.known.find(name);
366 if (val != m_compiler.m_types.known.end())
367 return *val->second;
368 } else if (const auto* qual_type = dyn_cast<ast::QualType>(&ast_type)) {
369 auto qualifier = qual_type->qualifier();
370 return convert_type(qual_type->base()).known_qual(qualifier);
371 } else if (isa<ast::SelfType>(ast_type)) {
372 if (parent != nullptr)
373 return *parent;
374 }
375
376 throw std::runtime_error("Cannot convert AST type to actual type!");
377};
