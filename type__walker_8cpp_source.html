34inline auto try_implicit_conversion(ast::OptionalExpr& expr, optional<ty::Type> target_ty) -> bool {
35 if (!target_ty)
36 return false;
37 if (!expr)
38 return false;
39
40 auto expr_ty = expr->ensure_ty();
41 auto compat = expr_ty.compatibility(*target_ty);
42 if (!compat.valid)
43 return false;
44
45 if (!compat.conv.empty())
46 wrap_in_implicit_cast(expr, compat.conv, target_ty);
47
48 return true;
49}

275void TypeWalker::direct_call_operator(ast::CallExpr& expr) {
276 yume_assert(expr.args.size() > 1, "Direct call must have at least 1 argument");
277 auto& base = *expr.args.front();
278 body_expression(base);
279
280 yume_assert(base.ensure_ty().base_isa<ty::Function>(), "Direct call target must be a function type");
281 const auto* base_ptr_ty = base.ensure_ty().base_cast<ty::Function>();
282

284 body_expression(*expr_arg);
285

287 yume_assert(compat.valid, "Invalid direct call with incompatible argument types");
288 if (compat.conv.empty())

291 wrap_in_implicit_cast(expr_arg, compat.conv, target);
292 }
293
294 expr.val_ty(base_ptr_ty->ret());

708auto TypeWalker::convert_type(ast::Type& ast_type) -> ty::Type {
709 auto parent = current_decl.self_ty();
710 const auto* context = static_cast<const DeclLike>(current_decl).subs();
711
712 if (const auto* simple_type = dyn_cast<ast::SimpleType>(&ast_type)) {
713 auto name = simple_type->name;
714 if (context != nullptr) {
715 auto generic = context->find(name);
716 if (generic != context->end())
717 return generic->second;
718 }
719 auto val = compiler.m_types.known.find(name);
720 if (val != compiler.m_types.known.end())
721 return val->second.get();
722 } else if (auto* qual_type = dyn_cast<ast::QualType>(&ast_type)) {
723 auto qualifier = qual_type->qualifier;
724 return convert_type(*qual_type->base).known_qual(qualifier);
725 } else if (isa<ast::SelfType>(ast_type)) {
726 if (parent) {
727 if (current_decl.opaque_self())
728 return {parent->known_opaque().base(), parent->is_mut(), parent->is_ref()};
729 return {parent->base(), parent->is_mut(), parent->is_ref()}; // TODO(rymiel): Isn't this just `parent`
730 }
731 } else if (auto* proxy_type = dyn_cast<ast::ProxyType>(&ast_type)) {
732 if (parent) {
733 if (const auto* parent_struct = parent->base_dyn_cast<ty::Struct>()) {
734 auto [target_type, target_offset] = find_field_ast(*parent_struct, proxy_type->field);
735 if (target_type != nullptr)
736 return convert_type(*target_type->raw_ptr());
737 throw std::runtime_error("Proxy type doesn't refer to a valid field?");
738 }
739 }
740 } else if (auto* fn_type = dyn_cast<ast::FunctionType>(&ast_type)) {
741 auto ret = optional<ty::Type>{};
742 auto args = vector<ty::Type>{};
743 if (auto& ast_ret = fn_type->ret; ast_ret.has_value())
744 ret = convert_type(*ast_ret);
745
746 for (auto& ast_arg : fn_type->args)
747 args.push_back(convert_type(*ast_arg));
748
749 if (fn_type->fn_ptr)
750 return compiler.m_types.find_or_create_fn_ptr_type(args, ret);
751 return compiler.m_types.find_or_create_fn_type(args, ret, {});
752 } else if (auto* templated = dyn_cast<ast::TemplatedType>(&ast_type)) {
753 auto& template_base = *templated->base;
754 expression(template_base);
755 auto base_type = convert_type(template_base);
756 auto* struct_obj = base_type.base_cast<ty::Struct>()->decl();
757
758 if (struct_obj == nullptr)
759 throw std::logic_error("Can't add template arguments to non-struct types");
760
761 for (auto& i : templated->type_args)
762 expression(*i);
763
764 Substitution subs = {};
765 for (const auto& [gen, gen_sub] : llvm::zip(struct_obj->type_args, templated->type_args))
766 subs.try_emplace(gen->name(), gen_sub->ensure_ty());
767
768 return get_or_declare_instantiation(struct_obj, subs);
769 }
770
771 throw std::runtime_error("Cannot convert AST type to actual type! "s + ast_type.kind_name() + " (" +
772 ast_type.describe() + ")");
773}

775void TypeWalker::resolve_queue() {
776 while (!decl_queue.empty()) {
777 auto next = decl_queue.front();
778 decl_queue.pop();
779
780 with_saved_scope([&] {
781 next.visit([](std::monostate /* absent */) {}, //
782 [](Const* /* cn */) {},
783 [&](Fn* fn) {
784 in_depth = true;
785 compiler.declare(*fn);
786 },
787 [&](Struct* st) {
788 for (auto& i : st->body()) {
789 auto decl = compiler.decl_statement(*i, st->self_ty, st->member);
790
791 // "Inherit" substitutions
792 // TODO(rymiel): Shadowing type variables should be an error
793 for (auto& [k, v] : st->subs)
794 decl.subs()->try_emplace(k, v);
795 compiler.walk_types(decl);
796 }
797 });
798 });
799 }
800}
