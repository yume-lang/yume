88template <> void TypeWalker::expression(ast::FieldAccessExpr& expr) {
89 body_expression(expr.base());
90 const auto& type = *expr.base().val_ty();
91
92 const auto* struct_type = dyn_cast<ty::Struct>(&type.without_qual());
93
94 if (struct_type == nullptr)
95 throw std::runtime_error("Can't access field of expression with non-struct type");
96
97 auto target_name = expr.field();
98 const ty::Type* target_type{};
99 int j = 0;
100 for (const auto& field : struct_type->fields()) {
101 if (field.name() == target_name) {
102 target_type = field.type().val_ty();
103 break;
104 }
105 j++;
106 }
107
108 expr.offset(j);
109 expr.val_ty(target_type);
110}

122template <> void TypeWalker::expression(ast::CallExpr& expr) {
123 auto overload_set = all_overloads_by_name(expr);
124 auto name = expr.name();
125
126 if (overload_set.empty())
127 throw std::logic_error("No function overload named "s + name);
128
129 for (auto& i : expr.args()) {
130 body_expression(i);
131 overload_set.arg_types.push_back(i.val_ty());
132 }
133
134#ifdef YUME_SPEW_OVERLOAD_SELECTION
135 errs() << "\n*** BEGIN OVERLOAD EVALUATION ***\n";
136 errs() << "Functions with matching names:\n";
137 overload_set.dump(errs());
138#endif
139
140 overload_set.determine_valid_overloads();
141
142#ifdef YUME_SPEW_OVERLOAD_SELECTION
143 errs() << "\nViable overloads:\n";
144 overload_set.dump(errs(), true);
145#endif
146
147 Overload best_overload = overload_set.best_viable_overload();
148
149#ifdef YUME_SPEW_OVERLOAD_SELECTION
150 errs() << "\nSelected overload:\n";
151 best_overload.dump(errs());
152 errs() << "\n*** END OVERLOAD EVALUATION ***\n\n";
153#endif
154
155 auto& instantiate = best_overload.instantiation;
156 auto& selected = best_overload.fn;
157
158 // It is an instantiation of a function template
159 if (!instantiate.sub.empty()) {
160 // TODO: move most of this logic directly into Fn
161 // Try to find an already existing instantiation with the same substitutions
162 auto existing_instantiation = m_current_fn->m_instantiations.find(instantiate);
163 if (existing_instantiation == m_current_fn->m_instantiations.end()) {
164 // An existing one wasn't found. Duplicate the template's AST with the substituted types.
165 auto* decl_clone = selected->m_ast_decl.clone();
166 m_current_fn->m_member->direct_body().emplace_back(decl_clone);
167
168 std::map<string, const ty::Type*> subs{};
169 for (const auto& [k, v] : instantiate.sub)
170 subs.try_emplace(k->name(), v);
171
172 auto fn_ptr = std::make_unique<Fn>(*decl_clone, selected->m_parent, selected->m_member, move(subs));
173 auto new_emplace = m_current_fn->m_instantiations.emplace(instantiate, move(fn_ptr));
174 auto& new_fn = *new_emplace.first->second;
175
176 // TODO: Push this in a queue to not cause a huge stack trace when an instantiation causes another instantiation
177 // TODO: This really needs to go into a queue because an instantiation can loop back to requiring to instantiate
178 // itself while the first instantiation hasn't finished existing this scope yet, leading to god knows what
179 // outcome.
180
181 // Save everything pertaining to the old context
182 auto saved_scope = m_scope;
183 auto* saved_current = m_current_fn;
184
185 // Temporarily create a new context
186 m_in_depth = false;
187 m_scope.clear();
188 m_current_fn = &new_fn;
189
190 // Type-annotate the cloned declaration
191 body_statement(*decl_clone);
192 m_in_depth = true;
193
194 auto* llvm_fn = m_compiler.declare(new_fn);
195 new_fn.m_llvm_fn = llvm_fn;
196 selected = &new_fn;
197
198 // Restore again
199 m_scope = saved_scope;
200 m_current_fn = saved_current;
201 } else {
202 selected = existing_instantiation->second.get();
203 }
204 }
205
206 for (auto [target, expr_arg, compat] :
207 llvm::zip(selected->ast().args(), expr.direct_args(), best_overload.compatibilities)) {
208 yume_assert(compat.valid, "Invalid compatibility after overload already selected?????");
209 if (compat.conv.empty())
210 continue;
211
212 const auto* target_type = target.type().val_ty();
213 auto cast_expr = std::make_unique<ast::ImplicitCastExpr>(expr_arg->token_range(), std::move(expr_arg), compat.conv);
214 cast_expr->val_ty(target_type);
215 expr_arg = std::move(cast_expr);
216 }
217
218 if (selected->ast().ret().has_value())
219 expr.val_ty(selected->ast().ret()->get().val_ty());
220
221 expr.selected_overload(selected);
222}

298auto TypeWalker::convert_type(const ast::Type& ast_type) -> const ty::Type& {
299 const ty::Type* parent = m_current_fn == nullptr ? m_current_struct : m_current_fn->parent();
300 Fn* context = m_current_fn;
301
302 if (const auto* simple_type = dyn_cast<ast::SimpleType>(&ast_type)) {
303 auto name = simple_type->name();
304 if (context != nullptr) {
305 auto generic = context->m_subs.find(name);
306 if (generic != context->m_subs.end())
307 return *generic->second;
308 }
309 auto val = m_compiler.m_types.known.find(name);
310 if (val != m_compiler.m_types.known.end())
311 return *val->second;
312 } else if (const auto* qual_type = dyn_cast<ast::QualType>(&ast_type)) {
313 auto qualifier = qual_type->qualifier();
314 return convert_type(qual_type->base()).known_qual(qualifier);
315 } else if (isa<ast::SelfType>(ast_type)) {
316 if (parent != nullptr)
317 return *parent;
318 }
319
320 throw std::runtime_error("Cannot convert AST type to actual type!");
321}
