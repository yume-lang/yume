260template <> void TypeWalker::expression(ast::CallExpr& expr) {
261 resolve_queue();
262
263 auto overload_set = all_fn_overloads_by_name(expr);
264 auto name = expr.name();
265
266 if (overload_set.empty())
267 throw std::logic_error("No function overload named "s + name);
268
269 for (auto& i : expr.args()) {
270 body_expression(*i);
271 overload_set.args.push_back(&*i);
272 }
273
274#ifdef YUME_SPEW_OVERLOAD_SELECTION
275 errs() << "\n*** BEGIN FN OVERLOAD EVALUATION ***\n";
276 errs() << "Functions with matching names:\n";
277 overload_set.dump(errs());
278#endif
279
280 overload_set.determine_valid_overloads();
281
282#ifdef YUME_SPEW_OVERLOAD_SELECTION
283 errs() << "\nViable overloads:\n";
284 overload_set.dump(errs(), true);
285#endif
286
287 Overload best_overload = overload_set.best_viable_overload();
288
289#ifdef YUME_SPEW_OVERLOAD_SELECTION
290 errs() << "\nSelected overload:\n";
291 best_overload.dump(errs());
292 errs() << "\n*** END FN OVERLOAD EVALUATION ***\n\n";
293#endif
294
295 auto& instantiate = best_overload.instantiation;
296 auto* selected = best_overload.fn;
297
298 // It is an instantiation of a function template
299 if (!instantiate.sub.empty()) {
300 // Try to find an already existing instantiation with the same substitutions
301 auto [already_existed, inst_fn] = selected->get_or_create_instantiation(instantiate);
302 if (!already_existed) {
303 // An existing one wasn't found. We've been given a duplicate of the template's AST but without types
304 // The duplicate will have its types set again according to the substitutions being used.
305 auto& new_fn = inst_fn;
306
307 // The types of the instantiated function must be set immediately (i.e. with in_depth = false)
308 // This is because the implicit cast logic below depends on the direct type being set here and bound type
309 // information doesn't propagate across ImplicitCastExpr...
310 // TODO(rymiel): Find a better solution; such as moving cast logic also into queue?
311 // However that would require evaluation of the queue very eagerly (i.e. immediately when the function is used,
312 // which kinda defeats the purpose of the queue). So I guess we'll just keep this until I think of a better
313 // solution
314 // TODO(rymiel): find a better solution other than the solution proposed above
315 with_saved_scope([&] {
316 in_depth = false;
317 current_decl = &new_fn;
318 body_statement(new_fn.ast());
319 });
320
321 decl_queue.push(&new_fn);
322
323 selected = &new_fn;
324 } else {
325 selected = &inst_fn;
326 }
327 }
328
329 for (auto [target, expr_arg, compat] :
330 llvm::zip(selected->ast().args(), expr.args(), best_overload.compatibilities)) {
331 yume_assert(compat.valid, "Invalid compatibility after overload already selected?????");
332 if (compat.conv.empty())
333 continue;
334
335 wrap_in_implicit_cast(expr_arg.unwrap(), compat.conv, target.type().val_ty());
336 }
337
338 // Find excess variadic arguments. Logic will probably change later, but for now, always pass by value
339 // TODO(rymiel): revisit
340 for (const auto& expr_arg : llvm::enumerate(expr.args())) {
341 if (expr_arg.index() >= selected->ast().args().size() && expr_arg.value()->val_ty()->is_mut()) {
342 const auto* target_type = expr_arg.value()->val_ty()->qual_base();
343 wrap_in_implicit_cast(expr_arg.value().unwrap(), ty::Conv{.dereference = true}, target_type);
344 }
345 }
346
347 if (selected->ast().ret().has_value())
348 expr.attach_to(&selected->ast());
349
350 expr.selected_overload(selected);
351}

474auto TypeWalker::convert_type(const ast::Type& ast_type) -> const ty::Type& {
475 const ty::Type* parent = current_decl.self_ty();
476 const auto* context = current_decl.subs();
477
478 if (const auto* simple_type = dyn_cast<ast::SimpleType>(&ast_type)) {
479 auto name = simple_type->name();
480 if (context != nullptr) {
481 auto generic = context->find(name);
482 if (generic != context->end())
483 return *generic->second;
484 }
485 auto val = compiler.m_types.known.find(name);
486 if (val != compiler.m_types.known.end())
487 return *val->second;
488 } else if (const auto* qual_type = dyn_cast<ast::QualType>(&ast_type)) {
489 auto qualifier = qual_type->qualifier();
490 return convert_type(qual_type->base()).known_qual(qualifier);
491 } else if (isa<ast::SelfType>(ast_type)) {
492 if (parent != nullptr)
493 return *parent;
494 }
495
496 throw std::runtime_error("Cannot convert AST type to actual type! "s + ast_type.kind_name() + " (" +
497 ast_type.describe() + ")");
498}

500void TypeWalker::resolve_queue() {
501 while (!decl_queue.empty()) {
502 auto next = decl_queue.front();
503 decl_queue.pop();
504
505 with_saved_scope([&] {
506 next.visit_decl(
507 [&](Fn* fn) {
508 in_depth = true;
509 compiler.declare(*fn);
510 },
511 [&](Struct* st) {
512 for (auto& i : st->body().body()) {
513 auto decl = compiler.decl_statement(*i, st->self_ty, st->member);
514 compiler.walk_types(decl);
515 }
516 },
517 [&](Ctor* /*ctor*/) { throw std::runtime_error("Unimplemented"); });
518 });
519 }
