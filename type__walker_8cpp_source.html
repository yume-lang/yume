60template <> void TypeWalker::expression(ast::CtorExpr& expr) {
61 for (auto& i : expr.args()) {
62 body_expression(i);
63 }
64
65 if (auto* templated = dyn_cast<ast::TemplatedType>(&expr.type())) {
66 auto& template_base = templated->base();

68 const auto& base_type = convert_type(template_base);
69 auto struct_obj =
70 std::ranges::find_if(m_compiler.m_structs, [&](const Struct& st) { return st.m_type == &base_type; });
71
72 if (struct_obj == m_compiler.m_structs.end())
73 throw std::logic_error("Can't add template arguments to non-struct types");
74
75 for (auto& i : templated->type_vars())
76 expression(*i);
77
78 // XXX: Duplicated from function overload handling
79 Instantiation instantiation = {};
80 for (const auto& [gen, gen_sub] : llvm::zip(struct_obj->m_type_args, templated->type_vars()))
81 instantiation.sub.try_emplace(gen.get(), gen_sub->val_ty());
82
83 auto [already_existed, inst_struct] = struct_obj->get_or_create_instantiation(instantiation);
84
85 if (!already_existed) {
86 auto& new_st = inst_struct;

89 m_in_depth = false;

92 body_statement(new_st.ast());
93 });

96 std::string name_with_types = new_st.ast().name() + templated->describe();

99 Compiler::create_struct(new_st.ast(), name_with_types));

102 m_decl_queue.push(&new_st);

134template <> void TypeWalker::expression(ast::FieldAccessExpr& expr) {
135 body_expression(expr.base());
136 const auto& type = *expr.base().val_ty();
137
138 const auto* struct_type = dyn_cast<ty::Struct>(&type.without_qual());

151 j++;
152 }

154 expr.offset(j);

168template <> void TypeWalker::expression(ast::CallExpr& expr) {
169 resolve_queue();
170
171 auto overload_set = all_overloads_by_name(expr);
172 auto name = expr.name();
173
174 if (overload_set.empty())
175 throw std::logic_error("No function overload named "s + name);
176
177 for (auto& i : expr.args()) {
178 body_expression(i);
179 overload_set.arg_types.push_back(i.val_ty());
180 }
181
182#ifdef YUME_SPEW_OVERLOAD_SELECTION
183 errs() << "\n*** BEGIN OVERLOAD EVALUATION ***\n";
184 errs() << "Functions with matching names:\n";
185 overload_set.dump(errs());
186#endif
187
188 overload_set.determine_valid_overloads();
189
190#ifdef YUME_SPEW_OVERLOAD_SELECTION
191 errs() << "\nViable overloads:\n";
192 overload_set.dump(errs(), true);
193#endif
194
195 Overload best_overload = overload_set.best_viable_overload();
196
197#ifdef YUME_SPEW_OVERLOAD_SELECTION
198 errs() << "\nSelected overload:\n";
199 best_overload.dump(errs());
200 errs() << "\n*** END OVERLOAD EVALUATION ***\n\n";
201#endif
202
203 auto& instantiate = best_overload.instantiation;
204 auto& selected = best_overload.fn;
205
206 // It is an instantiation of a function template
207 if (!instantiate.sub.empty()) {
208 // Try to find an already existing instantiation with the same substitutions
209 auto [already_existed, inst_fn] = selected->get_or_create_instantiation(instantiate);
210 if (!already_existed) {
211 // An existing one wasn't found. We've been given a duplicate of the template's AST but without types
212 // The duplicate will have its types set again according to the substitutions being used.
213 auto& new_fn = inst_fn;
214
215 // The types of the instantiated function must be set immediately (i.e. with in_depth = false)
216 // This is because the implicit cast logic below depends on the direct type being set here and bound type
217 // information doesn't propagate across ImplicitCastExpr...
218 // TODO: Find a better solution; such as moving cast logic also into queue?
219 // However that would require evaluation of the queue very eagerly (i.e. immediately when the function is used,
220 // which kinda defeats the purpose of the queue). So I guess we'll just keep this until I think of a better
221 // solution
222 // TODO: find a better solution other than the solution proposed above
223 with_saved_scope([&] {
224 m_in_depth = false;
225 m_current_fn = &new_fn;
226 body_statement(new_fn.ast());
227 });
228
229 m_decl_queue.push(&new_fn);
230
231 selected = &new_fn;
232 } else {
233 selected = &inst_fn;
234 }
235 }
236
237 for (auto [target, expr_arg, compat] :
238 llvm::zip(selected->ast().args(), expr.direct_args(), best_overload.compatibilities)) {
239 yume_assert(compat.valid, "Invalid compatibility after overload already selected?????");
240 if (compat.conv.empty())
241 continue;
242
243 const auto* target_type = target.type().val_ty();
244 auto cast_expr = std::make_unique<ast::ImplicitCastExpr>(expr_arg->token_range(), move(expr_arg), compat.conv);
245 cast_expr->val_ty(target_type);
246 expr_arg = move(cast_expr);
247 }
248
249 if (selected->ast().ret().has_value())
250 expr.attach_to(&selected->ast());
251
252 expr.selected_overload(selected);
253}

330auto TypeWalker::convert_type(const ast::Type& ast_type) -> const ty::Type& {
331 const ty::Type* parent = m_current_fn == nullptr ? m_current_struct->type() : m_current_fn->parent();
332 auto* context = m_current_fn == nullptr ? m_current_struct == nullptr ? nullptr : &m_current_struct->m_subs
333 : &m_current_fn->m_subs;
334
335 if (const auto* simple_type = dyn_cast<ast::SimpleType>(&ast_type)) {
336 auto name = simple_type->name();
337 if (context != nullptr) {
338 auto generic = context->find(name);
339 if (generic != context->end())
340 return *generic->second;
341 }
342 auto val = m_compiler.m_types.known.find(name);
343 if (val != m_compiler.m_types.known.end())
344 return *val->second;
345 } else if (const auto* qual_type = dyn_cast<ast::QualType>(&ast_type)) {
346 auto qualifier = qual_type->qualifier();
347 return convert_type(qual_type->base()).known_qual(qualifier);
348 } else if (isa<ast::SelfType>(ast_type)) {
349 if (parent != nullptr)
350 return *parent;
351 }
352
353 throw std::runtime_error("Cannot convert AST type to actual type! "s + ast_type.kind_name() + " (" +
354 ast_type.describe() + ")");
355}

357void TypeWalker::resolve_queue() {
358 while (!m_decl_queue.empty()) {
359 auto next = m_decl_queue.front();
360 m_decl_queue.pop();
361
362 with_saved_scope([&] {
363 std::visit(DeclLikeVisitor{[&](Fn* fn) {
364 m_in_depth = true;
365 m_compiler.declare(*fn);
366 },
367 [&](Struct* st) {
368 for (auto& i : st->body().body()) {
369 auto decl = m_compiler.decl_statement(i, st->m_type, st->m_member);
370 m_compiler.walk_types(decl);
371 }
372 }},
373 next);
374 });
375 }
