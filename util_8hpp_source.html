34/// \brief A view over a range of pointer-like types, where every element is dereferenced
35///
36/// Creating a dereference view from a type such as `vector<unique_ptr<T>>` will give a range-like view, where each
37/// element is a const T&
38template <std::ranges::input_range T>
39requires pointer_like<typename std::iterator_traits<std::ranges::iterator_t<T>>::value_type>
40class dereference_view {
41private:
42 using Base = T;
43
44 const T& m_base;
45
46 struct Iterator {
47 friend dereference_view;
48
49 private:
50 using Parent = dereference_view;
51 using Base = dereference_view::Base;
52 using Base_iter = std::ranges::iterator_t<std::add_const_t<Base>>;
53
54 Base_iter m_current = Base_iter();
55 Parent* m_parent = nullptr;
56
57 public:
58 using difference_type = std::ranges::range_difference_t<Base>;
59
60 Iterator() requires std::default_initializable<Base_iter>
61 = default;
62
63 constexpr Iterator(Parent* parent, Base_iter current) : m_current(std::move(current)), m_parent(parent) {}
64
65 constexpr auto operator*() const -> decltype(auto) { return **m_current; }
66
67 constexpr auto operator*() -> decltype(auto) { return **m_current; }

71 constexpr auto operator->() -> decltype(auto) { return *m_current; }

73 constexpr auto operator++() -> Iterator& {
74 ++m_current;

81 auto tmp = *this;
82 ++*this;
83 return tmp;

87 --m_current;

92 auto tmp = *this;
93 --*this;
94 return tmp;

98 m_current += n;
99 return *this;
100 }

103 m_current -= n;
104 return *this;
105 }

107 constexpr auto operator[](difference_type n) const
108 -> decltype(auto) requires std::ranges::random_access_range<Base> {
109 return *m_current[n];
110 }

112 friend constexpr auto operator==(const Iterator& x, const Iterator& y)
113 -> bool requires std::equality_comparable<Base_iter> {
114 return x.m_current == y.m_current;
115 }

122 }

131 return {i.m_parent, i.m_current - n};
132 }

135 return x.m_current - y.m_current;
136 }
137 };

154/// \brief Dereference an `optional` holding a pointer-like value, if it is holding a value
155/// Because C++20 doesn't have monadic operations for optionals yet, which should be a basic, fundamental feature of an
156/// "Optional" type
157template <typename T>
158requires pointer_like<T>
159auto inline constexpr try_dereference(const std::optional<T>& opt) {
160 using U = std::reference_wrapper<std::remove_reference_t<decltype(*opt.value())>>;
161 if (opt.has_value()) {
162 return std::optional<U>(*opt.value());
163 }
164 return std::optional<U>{};
165}

167/// Opens a writeable stream to a file with the given filename relative to the current working directory.
168auto inline open_file(const char* filename) -> unique_ptr<llvm::raw_pwrite_stream> {
169 std::error_code errorCode;
170 auto dest =
171 std::make_unique<llvm::raw_fd_ostream>(filename, errorCode, llvm::sys::fs::CreationDisposition::CD_CreateAlways);
172
173 if (errorCode) {
174 llvm::errs() << "Could not open file: " << errorCode.message() << "\n";
175 throw;
176 }

181/// \brief The stem of a path-like string, which is the component after the last slash.
182///
183/// "foo/bar/file.txt" -> "file.txt"
184[[nodiscard]] auto inline stem(std::string_view sv) -> std::string_view {
185 auto delim = sv.rfind('/');
186 return sv.substr(delim == string::npos ? 0 : delim + 1);
187}

189/// `Atom`s represent strings in a string pool.
190/** This means two atoms created with the same string value will contain
191 * identical pointers to that string. Thus, comparing `Atom`s is extremely cheap, as it only consists of a pointer
192 * equality check.
193 */
194struct Atom {
195 const string* m_str;
196
197 constexpr Atom() = delete;
198 explicit Atom(const string* str) : m_str{str} {}
199
200 operator std::string() const { // NOLINT(google-explicit-constructor)
201 return *m_str;
202 }
203 auto inline operator<=>(const Atom& other) const noexcept = default;
204
205 static auto inline make_atom(const string& value) noexcept -> Atom {
206 auto data = Atom::interned.emplace(value).first;
207 return Atom{&*data};
208 }
209
210private:
211 static inline std::set<string> interned; // NOLINT(cppcoreguidelines-avoid-non-const-global-variables)
212};
