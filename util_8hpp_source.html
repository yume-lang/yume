158template <typename T>
159requires pointer_like<T>
160auto inline constexpr try_dereference(const std::optional<T>& opt) {
161 using U = std::reference_wrapper<std::remove_reference_t<decltype(*opt.value())>>;
162 if (opt.has_value()) {
163 return std::optional<U>(*opt.value());
164 }
165 return std::optional<U>{};
166}
