200 [[nodiscard]] auto subs() const noexcept -> const Substitution* {
201 return visit([](std::monostate /*absent*/) noexcept -> const Substitution* { return nullptr; },
202 [](Const* /*cn*/) noexcept -> const Substitution* { return nullptr; },
203 [](auto* decl) noexcept -> const Substitution* {
204 if (decl == nullptr)
205 return nullptr;
206 auto& subs = decl->get_subs();
