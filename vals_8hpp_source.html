157/// A common base between declarations in the compiler: `Fn`, `Struct`, and `Ctor`. Its value may also be absent
158/// (`std::monostate`).
159struct DeclLike : public DeclLike_t {
160private:
161 template <typename R, typename... Ts> struct DeclLikeVisitor : Ts... {
162 using Ts::operator()...;
163 auto operator()(std::monostate /* ignored */) -> R { return R(); };
164 };
165 template <typename R, typename... Ts> DeclLikeVisitor(R, Ts...) -> DeclLikeVisitor<R, Ts...>;
166
167public:
168 using DeclLike_t::DeclLike_t;
169
170 template <typename R = void, typename... Ts> auto visit_decl(Ts... ts) {
171 return std::visit(DeclLikeVisitor<R, Ts...>{ts...}, *this);
172 }
173 template <typename R = void, typename... Ts> [[nodiscard]] auto visit_decl(Ts... ts) const {
174 return std::visit(DeclLikeVisitor<R, Ts...>{ts...}, *this);
175 }
176
177 [[nodiscard]] auto subs() const -> const substitution_t* {
178 return visit_decl<substitution_t*>([](Ctor* /*decl*/) -> substitution_t* { return nullptr; },
179 [](auto* decl) -> substitution_t* {
180 if (decl == nullptr)
181 return nullptr;
182 return &decl->subs;
183 });
184 };

186 [[nodiscard]] auto ast() const -> const ast::AST* {
187 return visit_decl<const ast::AST*>([](const auto* decl) -> const ast::AST* { return &decl->ast(); });
