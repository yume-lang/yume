45/// A function declaration in the compiler.
46/**
47 * The primary use of this struct is to bind together the AST declaration of a function (`ast::FnDecl`) and the bytecode
48 * body of a function (`llvm::Function`).
49 * A function template may also be a `Fn`. Since a template doesn't actually have a body, its `m_llvm_fn` is `nullptr`.
50 * All the instantiations of a template are stored in `m_instantiations`.
51 */
52struct Fn {
53 ast::FnDecl& m_ast_decl;
54 /// If this function is in the body of a struct, this points to its type. Used for the `self` type.
55 ty::Type* m_parent{};
56 /// The program this declaration is a member of.
57 ast::Program* m_member{};
58 vector<unique_ptr<ty::Generic>> m_type_args{};
59 /// If this is an instantiation of a template, a mapping between type variables and their substitutions.
60 std::map<string, const ty::Type*> m_subs{};
61 std::map<Instantiation, unique_ptr<Fn>> m_instantiations{};
62 llvm::Function* m_llvm_fn{};
63 /// A basic block where are allocations for local variables should go. It is placed \e before the "entrypoint".
