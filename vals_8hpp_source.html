98 template <typename... Ts>
99 auto visit_decl(std::invocable<ast::FnDecl&, Ts...> auto fn_c, std::invocable<ast::CtorDecl&, Ts...> auto ct_c,
100 Ts... ts) const -> decltype(auto) {
101 if (auto* fn_decl = dyn_cast<ast::FnDecl>(&ast_decl)) {
102 return fn_c(*fn_decl, std::forward<Ts...>(ts)...);
103 }
104 return ct_c(cast<ast::CtorDecl>(ast_decl), std::forward<Ts...>(ts)...);

112 for (auto& i : fn_decl->args())
113 vec.emplace_back(std::move<T>(fn(i)));
114 } else {
115 for (auto& i : cast<ast::CtorDecl>(ast_decl).args())
116 vec.emplace_back(std::move<T>(fn(i)));
117 }
118 return vec;

161/// A common base between declarations in the compiler: `Fn` and `Struct`. Its value may also be absent
162/// (`std::monostate`).
163struct DeclLike : public DeclLike_t {
164private:
165 template <typename R, typename... Ts> struct DeclLikeVisitor : Ts... {
166 using Ts::operator()...;
167 auto operator()(std::monostate /* ignored */) noexcept(noexcept(R())) -> R { return R(); };
168 };
169 template <typename R, typename... Ts> DeclLikeVisitor(R, Ts...) -> DeclLikeVisitor<R, Ts...>;
170
171public:
172 using DeclLike_t::DeclLike_t;
173

175 return std::visit(DeclLikeVisitor<R, Ts...>{ts...}, *this);
176 }
177 template <typename R = void, typename... Ts> [[nodiscard]] auto visit_decl(Ts... ts) const {
178 return std::visit(DeclLikeVisitor<R, Ts...>{ts...}, *this);
179 }

181 [[nodiscard]] auto subs() const noexcept -> const Substitution* {
182 return visit_decl<const Substitution*>([](auto* decl) noexcept -> const Substitution* {
183 if (decl == nullptr)
184 return nullptr;
185 auto& subs = decl->get_subs();
186 if (auto self = decl->get_self_ty(); subs.empty() && self.has_value())
187 if (auto* self_st = self->template base_dyn_cast<ty::Struct>())
188 return &self_st->subs();

190 return &subs;
191 });
192 };
