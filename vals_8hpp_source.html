26/// Common values between function declarations (`Fn`) and constructors (`Ctor`), which behave very similarly.
27struct FnBase {
28 /// The ast node that defines this declaration, being `ast::FnDecl` for `Fn` and `ast::CtorDecl` for `Ctor`. Should
29 /// not be accessed directly, instead \see ast() on `Fn` and `Ctor`.
30 ast::AST& ast;
31 /// If this function is in the body of a struct, this points to its type. Used for the `self` type.
32 optional<ty::Type> self_ty{};
33 /// The program this declaration is a member of.
34 ast::Program* member{};
35 /// If this is an instantiation of a template, a mapping between type variables and their substitutions.
36 Substitution subs{};
37 /// The LLVM function definition corresponding to this function or constructor.
38 llvm::Function* llvm{};
39
40 operator llvm::Function*() const { return llvm; }
41};

149 template <typename R, typename... Ts> struct DeclLikeVisitor : Ts... {
150 using Ts::operator()...;
151 auto operator()(std::monostate /* ignored */) -> R { return R(); };

155public:
